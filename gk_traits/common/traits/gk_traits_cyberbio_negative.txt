# Sign-flipped mirror of trait_seasonal_dormancy
# (15_biogenesis_species_traits.txt:132-144). Realised cost is 0.15 employed /
# 0.75 idle; the triggered block carries 0.60 because the flat block stacks.
trait_gk_sleepless = {
	cost = -2
	category = normal
	inline_script = "traits/gk_access_negative_machine"
	opposites = { "trait_seasonal_dormancy" "trait_low_maintenance" }
	allowed_archetypes = { BIOLOGICAL LITHOID MACHINE ROBOT }
	species_class = { AQUATIC ART }
	inline_script = "traits/gk_portraits_cyber"
	inline_script = "traits/gk_class_override_cyber"
	localized_tags = { SPECIES_TRAIT_AQUATIC SPECIES_TRAIT_ARTHROPOID SPECIES_TRAIT_GK_CYBER }
	tags = { organic negative upkeep aquatic arthropoid cybernetic species }
	custom_tooltip = trait_gk_sleepless_effect
	modifier = {
		planet_pops_upkeep_mult = 0.15
	}
	triggered_pop_group_modifier = {
		potential = {
			OR = {
				is_unemployed = yes
				is_pop_category = civilian
				is_pop_category = maintenance_drone
			}
		}
		planet_pops_upkeep_mult = 0.60
	}
	slave_cost = {
		trade = -200
	}
}
