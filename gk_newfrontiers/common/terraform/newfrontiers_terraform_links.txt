## Machine World ##
###################

terraform_link = {
	to = "pc_machine"
	from = "pc_sterile"
	
	resources = {
		category = terraforming

		cost = {
			energy = @machineCost
		}
	}
	
	duration = @machineTime
	
	potential = {
		has_ascension_perk = ap_machine_worlds
		NOT = { has_global_flag = has_real_terraforming_mod }
	}
	
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0
			from = {
				any_owned_pop_group = {
					NOR = {
						has_trait = trait_machine_unit
						has_trait = trait_mechanical
					}
				}
			}
		}
		modifier = {
			factor = 0
			from = {
				is_colony = no
			}
		}		
	}
}
