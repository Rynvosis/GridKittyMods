set_dune_subclass = {
	if = {
		limit = { NOT = { is_planet_class = pc_desert } }
		change_pc = pc_desert
	}
	remove_subclass_modifiers = yes
	set_planet_entity = {
		picture = pc_dune
		entity = dune_planet_01_entity
	}
	add_modifier = {
		modifier = "dune_world"
		days = -1
	}
	if = {
		limit = { NOT = { has_deposit = d_dust_desert } }
		add_deposit = d_dust_desert
	}
}
