# Sterile World signature district. Modelled on district_polytechnic.
district_vault = {
	expansion_planner = no
	base_buildtime = 250
	icon = district_nexus
	overlay_icon = GFX_district_amenities
	is_uncapped = { always = no }
	shared_capacity_modifier = district_vault

	show_tech_unlock_if = {
		is_nomadic = no
		is_wilderness_empire = no
	}

	allow = {
		custom_tooltip = {
			fail_text = arcology_project_construction_fail_tt
			NOT = { has_carrier_flag = arcology_project_construction }
		}
	}

	zone_slots = {
		slot_vault
	}

	show_on_uncolonized = { is_planet_class = pc_sterile }
	potential = { is_planet_class = pc_sterile }

	conversion_ratio = 1
	convert_to = {
		district_city
		district_hive
		district_nexus
	}

	resources = {
		category = planet_districts
		cost = { minerals = 450 }
		upkeep = { energy = 3 }
	}

	# storage is promised by the specializations, not the base district
	planet_modifier = {
		planet_housing_add = 200
	}

	inline_script = {
		script = jobs/clerks_add
		AMOUNT = 50
	}

	inline_script = { script = districts/gk_vault_triggered_name }

	ai_weight_coefficient = 1.5
	additional_ai_weight = 100
}
