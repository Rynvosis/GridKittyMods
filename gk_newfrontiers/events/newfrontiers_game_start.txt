namespace = newfrontiers

### Generate Starting Subclass
event = {
	id = newfrontiers.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		# Start Subclass Menu
		every_country = {
			limit = { is_ai = no }
			country_event = { id = newfrontiers.8 }
		}
		
		every_country = {
			limit = { 
				is_ai = yes
				OR = {
					is_country_type = default
					#is_country_type = fallen_empire
					is_country_type = primitive
				}
			}					
			capital_scope = { carrier_event = { id = newfrontiers.15 } }
		}
		
		#Living Sea
		random_galaxy_planet = {
			limit = { has_modifier = "living_sea" }
			set_planet_entity = {
				picture = pc_living_sea
				entity = ocean_planet_sea_of_conciousness_entity
			}
		}
	}
}
