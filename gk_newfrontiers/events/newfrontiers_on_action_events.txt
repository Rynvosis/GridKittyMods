carrier_event = {
	id = newfrontiers_action.2
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		THIS = {
			remove_subclass_modifiers = yes
		}
	}

carrier_event = {
	id = newfrontiers_action.13
	hide_window = yes

	is_triggered_only = yes
	
	trigger = {
		is_planet_subclass = yes
	}
	
	immediate = {
		remove_incorrect_subclass = yes
	}
}

### Nuclear war happened
carrier_event = {
	id = newfrontiers_action.14
	hide_window = yes

	is_triggered_only = yes
	
	trigger = {
		is_planet_class = pc_nuked
		is_planet_subclass = no
		NOT = { has_carrier_flag = ratling_planet }   # never convert the Ketling home worlds
	}

	immediate = {
		random_list = {
			30 = {
				set_fallout_subclass = yes
			}
			70 = {
				set_wasteland_subclass = yes
			}
		}
	}
}
