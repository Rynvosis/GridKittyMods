### Bioluminescent World

# Bioluminescent World colonized
carrier_event = {
	id = newfrontiers_action.16
	hide_window = yes

	is_triggered_only = yes
	
	trigger = {
		has_modifier = bioluminescent_world
	}
	
	immediate = {
		if = {
			limit = { 
				exists = owner
				owner = { is_regular_empire = yes }
				NOT = { has_modifier = preserve_planet }
			}
			carrier_event = { id = newfrontiers_action.17 days = 60 random = 30 }
		}
		if = {
			limit = { 
				exists = owner
				owner = { is_regular_empire = yes }
				has_modifier = preserve_planet
				num_buildings = {
					type = any
					value > 0
					category = manufacturing
				}
			}
			carrier_event = { id = newfrontiers_action.20 days = 5 }
		}
		if = {
			limit = { 
				has_modifier = environmentalists_protests
				num_buildings = {
					type = any
					value > 0
					category = manufacturing
				}
			}
			carrier_event = { id = newfrontiers_action.20 days = 5 }
		}
		if = {
			limit = { 
				exists = owner
				owner = { is_regular_empire = no }
				has_modifier = environmentalists_protests
			}
			remove_modifier = environmentalists_protests
		}
		carrier_event = { id = newfrontiers_action.16 days = 1800 random = 900 }
	}
}

carrier_event = {
	id = newfrontiers_action.17
	title = "newfrontiers_action.17.name"
	desc = "newfrontiers_action.17.desc"
	picture = GFX_evt_bioluminescent_world
	show_sound = event_alien_nature
	location = ROOT
	
	is_triggered_only = yes
	
	trigger = {
		has_modifier = bioluminescent_world
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = preserve_planet }
	}
	
	immediate = {
	}

	option = {
		name = "newfrontiers_action.17.a"
		add_modifier = {
			modifier = "preserve_planet"
			days = -1
		}
		hidden_effect = {
			every_deposit = {
				if = {
					limit = {
						OR = {
							is_deposit_type = d_mineral_fields
							is_deposit_type = d_prosperous_mesa
							is_deposit_type = d_submerged_ore_veins
						}
					}
					root = { 
						add_blocker = {
							type = d_nature_preserve
							blocked_deposit = prev
						}
					}
				}
			}
		}
	}
	
	option = {
		name = "newfrontiers_action.17.b"
		hidden_effect = {
			carrier_event = { id = newfrontiers_action.18 days = 10 random = 5 }
		}
	}
}
