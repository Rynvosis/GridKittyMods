# Overwrite: quantum_catapult bypass
# Vanilla gates the catapult on `prerequisites = { tech_quantum_catapult }`, which
# requires the *user* to have the tech. We drop the prerequisite and instead allow
# use if either party — the user or the system owner — has researched it. This
# lets allies launch through a catapult belonging to a teched-up partner without
# needing the tech themselves.

quantum_catapult = {
	name = quantum_catapult_BYPASS
	is_pathfind = no
	uses_action = no
	requires_exploration = no
	always_active = yes
	windup_time = 30
	winddown_time = 360
	ftl_multiplier = 0
	icon_frame = 25
	remember_instances = no
	galactic_map_tooltip_header = quantum_catapult_BYPASS
	galactic_map_tooltip_desc = quantum_catapult_BYPASS_DESC
	connection_type = to_any_system
	animation_time = 1
	ftl_ship_effect = "gateway_ftl_ship_effect_entity"
	ftl_bypass_effect = "gateway_ftl_bypass_effect_entity"
	country_can_use = {
		custom_tooltip = {
			fail_text = "gk_utils_quantum_catapult_requires_tech"
			OR = {
				has_technology = tech_quantum_catapult
				AND = {
					exists = from
					from = { has_technology = tech_quantum_catapult }
				}
			}
		}
		custom_tooltip = {
			fail_text = QUANTUM_CATAPULT_COUNTRY_CANNOT_USE
			OR = {
				NOT = { exists = from }
				is_same_value = from
				is_in_federation_with = from
				is_overlord_to = from
				from = { is_overlord_to = root }
			}
		}
	}
	on_action = "on_entering_quantum_catapult"
}
