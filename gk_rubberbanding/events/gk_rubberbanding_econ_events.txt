####################################
# Utilization-Weighted Economy Rubberbanding
# Events
####################################
# Monthly event chain that calculates consumption-based economic penalties
# Goal: Penalize countries based on how much they CONSUME (utilization), weighted by resource value
# Formula: Penalty_r = utilization_r × Mal_total × EMA(Revenue_r)
# where utilization_r = EMA(Use_r) / EMA(Revenue_r)

namespace = gkrb_econ

####################################
# Event 1: Initialize Globals
####################################
# Init globals once on game start

event = {
	id = gkrb_econ.1
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		event_target:global_event_country = {
			if = {
				limit = {
					NOT = { has_country_flag = gkrb_econ_has_global_variables }
				}
				# Global aggregation variables
				set_variable = {
					which = gkrb_econ_global_econ_sum
					value = 0
				}
				set_variable = {
					which = gkrb_econ_global_country_count
					value = 0
				}
				set_variable = {
					which = gkrb_econ_global_econ_avg
					value = 1
				}
				
				# Load resource weights from scripted_variables
				# Basic resources
				set_variable = { which = gkrb_econ_weight_energy value = @gkrb_econ_weight_energy }
				set_variable = { which = gkrb_econ_weight_minerals value = @gkrb_econ_weight_minerals }
				set_variable = { which = gkrb_econ_weight_food value = @gkrb_econ_weight_food }
				set_variable = { which = gkrb_econ_weight_trade value = @gkrb_econ_weight_trade }
				
				# Advanced resources
				set_variable = { which = gkrb_econ_weight_consumer_goods value = @gkrb_econ_weight_consumer_goods }
				set_variable = { which = gkrb_econ_weight_alloys value = @gkrb_econ_weight_alloys }
				
				# Strategic resources
				set_variable = { which = gkrb_econ_weight_volatile_motes value = @gkrb_econ_weight_volatile_motes }
				set_variable = { which = gkrb_econ_weight_exotic_gases value = @gkrb_econ_weight_exotic_gases }
				set_variable = { which = gkrb_econ_weight_rare_crystals value = @gkrb_econ_weight_rare_crystals }
				
				# Exotic resources
				set_variable = { which = gkrb_econ_weight_sr_living_metal value = @gkrb_econ_weight_sr_living_metal }
				set_variable = { which = gkrb_econ_weight_sr_zro value = @gkrb_econ_weight_sr_zro }
				set_variable = { which = gkrb_econ_weight_sr_dark_matter value = @gkrb_econ_weight_sr_dark_matter }
				set_variable = { which = gkrb_econ_weight_nanites value = @gkrb_econ_weight_nanites }
				
				# Abstract resources
				set_variable = { which = gkrb_econ_weight_unity value = @gkrb_econ_weight_unity }
				set_variable = { which = gkrb_econ_weight_physics_research value = @gkrb_econ_weight_physics_research }
				set_variable = { which = gkrb_econ_weight_society_research value = @gkrb_econ_weight_society_research }
				set_variable = { which = gkrb_econ_weight_engineering_research value = @gkrb_econ_weight_engineering_research }
				
				set_country_flag = gkrb_econ_has_global_variables
			}
		}

		every_playable_country = {
			# initialize stockpile tracking variables
			set_variable = { which = gkrb_prev_stockpile_energy value = value:gkrb_current_stock|RESOURCE|energy| }
			set_variable = { which = gkrb_prev_stockpile_minerals value = value:gkrb_current_stock|RESOURCE|minerals| }
			set_variable = { which = gkrb_prev_stockpile_food value = value:gkrb_current_stock|RESOURCE|food| }
			set_variable = { which = gkrb_prev_stockpile_trade value = value:gkrb_current_stock|RESOURCE|trade| }
			set_variable = { which = gkrb_prev_stockpile_consumer_goods value = value:gkrb_current_stock|RESOURCE|consumer_goods| }
			set_variable = { which = gkrb_prev_stockpile_alloys value = value:gkrb_current_stock|RESOURCE|alloys| }
			set_variable = { which = gkrb_prev_stockpile_volatile_motes value = value:gkrb_current_stock|RESOURCE|volatile_motes| }
			set_variable = { which = gkrb_prev_stockpile_exotic_gases value = value:gkrb_current_stock|RESOURCE|exotic_gases| }
			set_variable = { which = gkrb_prev_stockpile_rare_crystals value = value:gkrb_current_stock|RESOURCE|rare_crystals| }
			set_variable = { which = gkrb_prev_stockpile_sr_living_metal value = value:gkrb_current_stock|RESOURCE|sr_living_metal| }
			set_variable = { which = gkrb_prev_stockpile_sr_zro value = value:gkrb_current_stock|RESOURCE|sr_zro| }
			set_variable = { which = gkrb_prev_stockpile_sr_dark_matter value = value:gkrb_current_stock|RESOURCE|sr_dark_matter| }
			set_variable = { which = gkrb_prev_stockpile_nanites value = value:gkrb_current_stock|RESOURCE|nanites|	}
			set_variable = { which = gkrb_prev_stockpile_unity value = value:gkrb_current_stock|RESOURCE|unity| }
			set_variable = { which = gkrb_prev_stockpile_physics_research value = value:gkrb_current_stock|RESOURCE|physics_research| }
			set_variable = { which = gkrb_prev_stockpile_society_research value = value:gkrb_current_stock|RESOURCE|society_research| }
			set_variable = { which = gkrb_prev_stockpile_engineering_research value = value:gkrb_current_stock|RESOURCE|engineering_research| }
		}
	}

}

# STAGE 5: Apply Economy Modifiers
####################################
# Event 50: Apply economy modifiers based on aheadness
# Calculate aheadness, apply penalties or bonuses

country_event = {
	id = gkrb_econ.50
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		# Always clear first
		gkrb_clear_econ_rubberbanding_modifiers = yes
		
		# Calculate aheadness ratio and penalties
		gkrb_calculate_penalties = yes
		
		# If behind (more than 5% below average), apply catch-up bonuses
		if = {
			limit = {
				check_variable = {
					which = gkrb_aheadness_ratio
					value < -0.05
				}
			}
			gkrb_apply_behind_bonuses = yes
		}
		# If ahead (more than 5% above average), apply penalties
		else_if = {
			limit = {
				check_variable = {
					which = gkrb_aheadness_ratio
					value > 0.05
				}
			}
			gkrb_apply_ahead_penalties = yes
		}
		# Neutral zone - show status without penalties
		else = {
			add_modifier = {
				modifier = gkrb_econ_neutral_mod
				days = -1
			}
		}
	}
}
