# revenue
gkrp_resource_revenue = {
	base = 0
	complex_trigger_modifier = {
		trigger = resource_revenue_compare
		parameters = {
			resource = $RESOURCE$
		}
		mode = add
	}
}

# maintenance (heuristic for how much was put back into production upkeeps
gkrb_maintenance = {
	add = value:gkrb_maintenance_category|RESOURCE|$RESOURCE$|CATEGORY|planets| # factors in everything on planets, pops, jobs, structures
	add = value:gkrb_maintenance_category|RESOURCE|$RESOURCE$|CATEGORY|megastructures|
	add = value:gkrb_maintenance_category|RESOURCE|$RESOURCE$|CATEGORY|stations|
}
gkrb_maintenance_category = {
	complex_trigger_modifier = {
		trigger = resource_expenses_compare
		parameters = {
			resource = $RESOURCE$
			category = $CATEGORY$
		}
		mode = add
	}
}
gkrb_maintenance_discount = {
	add = value:gkrb_maintenance|RESOURCE|$RESOURCE$|
	mult = @gkrb_maintenance_discount
}

gkrp_effective_resource_revenue = {
	add = value:gkrp_resource_revenue|RESOURCE|$RESOURCE$|
	subtract = value:gkrb_maintenance_discount|RESOURCE|$RESOURCE$|
}

# utilization (if we expect +100 alloys, but your stockpile went down 200, you used 300)
gkrb_resource_use = {
	# how much we expect your stockpile to change this month
	add = value:gkrp_effective_resource_revenue|RESOURCE|$RESOURCE$|

	# how much your stockpile actually changed since last month
	subtract = value:gkrp_delta_stock|RESOURCE|$RESOURCE$|
}
gkrp_current_stock = {
	complex_trigger_modifier = {
		trigger = resource_stockpile_compare
		parameters = {
			resource = $RESOURCE$
		}
		mode = add
	}
}
gkrp_delta_stock = {
	add = value:gkrp_current_stock|RESOURCE|$RESOURCE$|

	# subtract previous stockpile to get delta, if not set we just assume 0
	modifier = {
		NOT = { is_variable_set = gkrb_prev_stock_$RESOURCE$ }
		set = 0
	}
}
# Deprecated: use EMA-based gkrb_econ_utilization_ratio instead
gkrp_utilization_ratio = {
	add = value:gkrb_resource_use|RESOURCE|$RESOURCE$|
	divide = value:gkrp_effective_resource_revenue|RESOURCE|$RESOURCE$|
}

# Deprecated: use EMA-based gkrb_econ_effective_power_resource instead
gkrp_effective_power = {
	add = value:gkrb_econ_utilization_ratio|RESOURCE|$RESOURCE$|
	mult = gkrb_econ_ema_revenue_$RESOURCE$
}



####################################
# Bloc Pairwise Calculations
####################################

# Pairwise modifier: (my_power × partner_power) / ((my_power + partner_power)²)
# Requires gkrb_temp_my_power and gkrb_temp_partner_power to be set first

# Denominator = (my_power + partner_power)²
# Returns at least 1 to avoid division by zero
gkrb_bloc_pairwise_denominator = {
	base = 0
	add = gkrb_temp_my_power
	add = gkrb_temp_partner_power
	pow = 2
	min = 1
}

# Remaining capacity for diminishing aggregation: 1 - current_increase
gkrb_bloc_remaining_capacity = {
	base = 1
	subtract = gkrb_econ_bloc_power_increase
	min = 0
}

gkrb_bloc_power_increase = {
	# calculate the pairwise contribution
	base = 0
	add = gkrb_temp_my_power
	mult = gkrb_temp_partner_power
	divide = value:gkrb_bloc_pairwise_denominator
	min = 0

	# scale by type of bloc
	mult = $BLOC_MULT$

	# scale by remaining capacity
	mult = value:gkrb_bloc_remaining_capacity
}

####################################
# EMA (Exponential Moving Average)
####################################

# Effective half-life: min(months_elapsed, target_halflife)
# This makes the EMA more responsive in early months
gkrb_econ_effective_halflife = {
    base = 0
    add = event_target:global_event_country.gkrb_econ_months_elapsed
    min = 1  # Minimum of 1 to avoid division by zero
    max = @gkrb_econ_ema_halflife_months  # Cap at target half-life
}

gkrb_compute_exponent_from_cached_halflife = {
	base = 1
	divide = value:gkrb_econ_effective_halflife
}

# Helper: compute decay from cached halflife (used once per tick)
gkrb_compute_decay_from_cached_halflife = {
	base = 0.5
	pow = value:gkrb_compute_exponent_from_cached_halflife
}

# Decay = 0.5 ^ (1/halflife) - weight retained from previous score
# Cached in global_event_country to avoid repeated pow calculations
gkrb_econ_ema_decay = {
	base = 0
	add = event_target:global_event_country.gkrb_econ_ema_decay_cached
}

# Alpha = 1 - decay - weight given to new data each month
# Cached in global_event_country
gkrb_econ_ema_alpha = {
	base = 1
	subtract = event_target:global_event_country.gkrb_econ_ema_decay_cached
}

# Per-resource EMA update helpers (use cached decay/alpha)
gkrb_ema_update_revenue = {
	add = value:gkrb_ema_update_prev|TYPE|revenue|RESOURCE|$RESOURCE$|
	add = value:gkrb_ema_update_new_revenue|RESOURCE|$RESOURCE$|
}

gkrb_ema_update_use = {
	add = value:gkrb_ema_update_prev|TYPE|use|RESOURCE|$RESOURCE$|
	add = value:gkrb_ema_update_new_use|RESOURCE|$RESOURCE$|
}

gkrb_ema_update_prev = {
	add = gkrb_econ_ema_$TYPE$_$RESOURCE$
	mult = value:gkrb_econ_ema_decay
}

gkrb_ema_update_new_revenue = {
	add = value:gkrp_effective_resource_revenue|RESOURCE|$RESOURCE$|
	mult = value:gkrb_econ_ema_alpha
}

gkrb_ema_update_new_use = {
	add = value:gkrb_resource_use|RESOURCE|$RESOURCE$|
	mult = value:gkrb_econ_ema_alpha
}

# Utilization ratio from EMA windows (no caps)
gkrb_econ_utilization_ratio = {
	base = 0
	add = gkrb_econ_ema_use_$RESOURCE$
	modifier = { 
		AND = {
			check_variable_arithmetic = {
				which = value:gkrb_econ_abs_revenue|RESOURCE|$RESOURCE$|
				value >= 0.01
			}
		}
		divide = gkrb_econ_ema_revenue_$RESOURCE$
	}
	modifier = {
		check_variable_arithmetic = {
			which = value:gkrb_econ_abs_revenue|RESOURCE|$RESOURCE$|
			value < 0.01
		}
		set = 0
	}
}

gkrb_econ_abs_revenue = {
	add = gkrb_econ_ema_revenue_$RESOURCE$
	abs = yes
}

# Effective power from EMA: (0.5 + 0.5*utilization) × ema_revenue
gkrb_econ_effective_power_resource = {
	base = 0
	add = value:gkrb_econ_utilization_ratio|RESOURCE|$RESOURCE$|
	mult = 0.5
	add = 0.5
	mult = gkrb_econ_ema_revenue_$RESOURCE$
}


gkrb_econ_apply_ema = {
	base = 0
	add = value:gkrb_prev_ema
	add = value:gkrb_new_score
}

gkrb_prev_ema = {
	add = gkrb_econ_score
	mult = value:gkrb_econ_ema_decay
}

gkrb_new_score = {
	add = gkrb_econ_total_value
	mult = value:gkrb_econ_ema_alpha
}

####################################
# Aheadness Ratio Calculation
####################################
# AR_total = (effective_power / galaxy_avg_ec_total) - 1
# effective_power = gkrb_econ_score × (1 + bloc_increase)
# Positive = ahead (penalized), Negative = behind (bonus)

gkrb_aheadness_ratio = {
	base = 0
	add = gkrb_econ_effective_power
	divide = event_target:global_event_country.gkrb_econ_global_econ_avg
	subtract = 1
}

####################################
# Penalty/Bonus Multipliers
####################################

# Total penalty multiplier (for countries ahead of average)
# Mal_total = clamp(AR_total, 0, 1)²
# At 100% ahead (AR=1), penalty = 100%
# At 50% ahead (AR=0.5), penalty = 25%
# At 25% ahead (AR=0.25), penalty = 6.25%
gkrb_total_penalty_mult = {
	base = 0
	add = value:gkrb_aheadness_ratio
	min = 0
	max = 1
	pow = 2
}

# Behind bonus multiplier (for countries behind average)
# behind_mult = clamp(abs(AR_total), 0, 1)²
# At 50% behind (AR=-0.5), bonus mult = 0.25
# At 100% behind (AR=-1 or worse), bonus mult = 1.0
gkrb_behind_mult = {
	base = 0
	add = value:gkrb_aheadness_ratio
	# Negate to get positive value for "behindness"
	mult = -1
	min = 0
	max = 1
	pow = 2
}

# Legacy compatibility: gkrb_econ_mult now returns total_penalty_mult
gkrb_econ_mult = {
	base = 0
	add = value:gkrb_total_penalty_mult
}
