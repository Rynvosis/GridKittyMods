####################################
# Utilization-Weighted Economy Rubberbanding
# Scripted Effects
####################################
# This system calculates penalties based on CONSUMPTION not production.
# EconValue_r = utilization_r × EMA(Revenue_r) × weight_r
# where utilization_r = EMA(Use_r) / EMA(Revenue_r)
# Penalty_r = utilization_r × Mal_total × EMA(Revenue_r)

####################################
# Phase 0: Calculate Economic Value
####################################
# Calculate Use, maintenance, and EconValue for each resource
# Then calculate total economic value (EC_total)

gkrb_calculate_econ_values = {
	# Reset total economic value
	set_variable = {
		which = gkrb_econ_total_value
		value = 0
	}
	
	# Basic resources
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = energy }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = minerals }
	# log = "\\[GKRB] Phase0 empire=\\[This.GetName] resource=mineral ema_rev=\\[This.gkrb_econ_ema_revenue_minerals] ema_use=\\[This.gkrb_econ_ema_use_minerals] util=\\[This.gkrb_econ_utilization_ratio_minerals] eff_power=\\[This.gkrb_econ_effective_power_minerals] econ_value=\\[This.gkrb_econ_value_minerals] total=\\[This.gkrb_econ_total_value]"
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = food }
	
	# Advanced resources
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = consumer_goods }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = alloys }
	# Trade uses special handling due to Stellaris trade policy mechanics
	# inline_script = { script = gk_rubberbanding/calculate_trade_econ_value }
	set_variable = { which = gkrb_additional_maintenance_discount_trade value = value:gkrb_additional_maintenance_discount_trade }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = trade }
	clear_variable = gkrb_additional_maintenance_discount_trade

	# Strategic resources
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = volatile_motes }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = exotic_gases }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = rare_crystals }
	# Exotic resources
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = sr_living_metal }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = sr_zro }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = sr_dark_matter }
	inline_script = { script = gk_rubberbanding/calculate_resource_econ_value RESOURCE = nanites }
	
	# Abstract resources - track EMA revenue only (no utilization)
	# They have their own rubberbanding systems for penalties
	# But contribute to economy power when in a lead
	inline_script = { script = gk_rubberbanding/calculate_abstract_resource_ema_revenue RESOURCE = unity }
	inline_script = { script = gk_rubberbanding/calculate_abstract_resource_ema_revenue RESOURCE = physics_research }
	inline_script = { script = gk_rubberbanding/calculate_abstract_resource_ema_revenue RESOURCE = society_research }
	inline_script = { script = gk_rubberbanding/calculate_abstract_resource_ema_revenue RESOURCE = engineering_research }
	
	# Add weighted research contribution (EMA revenue × tech lead factor)
	change_variable = {
		which = gkrb_econ_total_value
		value = value:gkrb_research_contribution_to_econ
	}

	# Add weighted unity contribution (EMA revenue × 2 × unity lead factor)
	change_variable = {
		which = gkrb_econ_total_value
		value = value:gkrb_unity_contribution_to_econ
	}
}


####################################
# Phase 1: Finalize Score + EMA
####################################
# Blend raw EC_total into EMA (bloc applied later for penalty calculations)

gkrb_finalize_econ_score = {

	# Blend current EC_total into rolling EMA score
	create_variable_if_not_exists = {
		VARIABLE = gkrb_econ_score
		VALUE_IF_CREATED = gkrb_econ_total_value
	}
	set_variable = {
		which = gkrb_econ_score
		value = value:gkrb_econ_apply_ema
	}
}

# Contribute to global sum
gkrb_contribute_to_global_sum = {
	event_target:global_event_country = {
		change_variable = {
			which = gkrb_econ_global_econ_sum
			value = prev.gkrb_econ_score
		}
		change_variable = {
			which = gkrb_econ_global_country_count
			value = 1
		}
	}
}

####################################
# Phase 1.5: Bloc Power Inheritance
####################################
# Calculate bloc power increase from all bloc partners
# Uses pairwise formula with EC_total (gkrb_econ_score)

gkrb_calculate_bloc_power_increase = {
	# Reset bloc power increase
	set_variable = {
		which = gkrb_econ_bloc_power_increase
		value = 0
	}
	
	# Get my EMA score
	set_variable = {
		which = gkrb_temp_my_power
		value = gkrb_econ_score
	}

	every_playable_country = {
		limit = {
			NOT = { is_same_value = prev }
		}
		# Federation
		if = {
			limit = { is_in_federation_with = prev }
			prev = { gkrb_add_bloc_partner_power = {
				PARTNER = prevprev
				BLOC_MULT = @gkrb_bloc_mult_federation
			}}
		}
		# Defensive pact
		if = { 
			limit = { has_defensive_pact = prev }
			prev = { gkrb_add_bloc_partner_power = {
				PARTNER = prevprev
				BLOC_MULT = @gkrb_bloc_mult_defensive_pact
			}}
		}
		# You are their vassal
		if = {
			limit = { is_overlord_to = prev }
			prev = { gkrb_add_bloc_partner_power = {
				PARTNER = prevprev
				BLOC_MULT = @gkrb_bloc_mult_overlord
			}}
		}
		# They are your vassal
		if = {
			limit = { prev = { is_overlord_to = prevprev } }
			prev = { gkrb_add_bloc_partner_power = {
				PARTNER = prevprev
				BLOC_MULT = @gkrb_bloc_mult_vassal
			}}
		}
	}

	clear_variable = gkrb_temp_my_power
}

# Add a single bloc partner's contribution to power increase
gkrb_add_bloc_partner_power = {
	# Get partner's EMA score
	set_variable = {
		which = gkrb_temp_partner_power
		value = $PARTNER$.gkrb_econ_score
	}

	# Calculate pairwise contribution and add to bloc power increase
	change_variable = {
		which = gkrb_econ_bloc_power_increase
		value = value:gkrb_bloc_power_increase|BLOC_MULT|$BLOC_MULT$|
	}

	clear_variable = gkrb_temp_partner_power
}

####################################
# Phase 3: Calculate & Apply Penalties
####################################
# Calculate aheadness ratio, penalty multiplier, and apply penalties

gkrb_calculate_penalties = {
	# Calculate effective power: score × (1 + bloc_increase)
	set_variable = {
		which = gkrb_econ_effective_power
		value = gkrb_econ_score
	}
	if = {
		limit = { is_variable_set = gkrb_econ_bloc_power_increase }
		set_variable = {
			which = gkrb_temp_bloc_mult
			value = gkrb_econ_bloc_power_increase
		}
		change_variable = {
			which = gkrb_temp_bloc_mult
			value = 1
		}
		multiply_variable = {
			which = gkrb_econ_effective_power
			value = gkrb_temp_bloc_mult
		}
	}
	# Bloc multiplier is scratch-only
	clear_variable = gkrb_temp_bloc_mult
	
	# Calculate aheadness ratio: AR = (effective_power / avg) - 1
	set_variable = {
		which = gkrb_aheadness_ratio
		value = value:gkrb_aheadness_ratio
	}
	
	# Calculate total penalty mult: Mal = clamp(AR, 0, 1)²
	set_variable = {
		which = gkrb_total_penalty
		value = value:gkrb_total_penalty_mult
	}
	
	# Calculate behind bonus mult
	set_variable = {
		which = gkrb_behind_mult
		value = value:gkrb_behind_mult
	}
	
	# Calculate relative score for display
	set_variable = {
		which = gkrb_econ_relative
		value = value:gkrb_econ_relative_to_global
	}
	
	# For legacy compatibility
	set_variable = {
		which = gkrb_econ_mult
		value = gkrb_total_penalty
	}
}
