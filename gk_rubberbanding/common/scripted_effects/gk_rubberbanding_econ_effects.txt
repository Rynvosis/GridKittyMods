####################################
# Economy Score Calculations      #
####################################
# Calculate current economic power (snapshot) by summing weighted resource values
# Then blend into rolling EMA score for smoother rubberbanding
gkrb_calculate_econ_score = {
	# First calculate current power (snapshot)
	set_variable = {
		which = gkrb_econ_current_power
		value = 0
	}
	# Basic resources
	gkrb_add_weighted_resource_score = {
		RESOURCE = energy
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = minerals
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = food
	}
	# Advanced resources
	gkrb_add_weighted_resource_score = {
		RESOURCE = trade
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = consumer_goods
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = alloys
	}
	# Strategic resources
	gkrb_add_weighted_resource_score = {
		RESOURCE = volatile_motes
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = exotic_gases
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = rare_crystals
	}
	# Rare resources
	gkrb_add_weighted_resource_score = {
		RESOURCE = sr_living_metal
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = sr_zro
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = sr_dark_matter
	}
	# Unity and Research
	gkrb_add_weighted_resource_score = {
		RESOURCE = unity
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = physics_research
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = society_research
	}
	gkrb_add_weighted_resource_score = {
		RESOURCE = engineering_research
	}
	
	# Blend current power into rolling EMA score
	# Formula: score = score * decay + current_power * alpha
	# If score not initialized, use current_power directly
	if = {
		limit = { NOT = { is_variable_set = gkrb_econ_score } }
		set_variable = {
			which = gkrb_econ_score
			value = gkrb_econ_current_power
		}
	}
	else = {
		# score = score * decay
		multiply_variable = {
			which = gkrb_econ_score
			value = value:gkrb_econ_ema_decay
		}
		# temp = current_power * alpha
		set_variable = {
			which = gkrb_temp_ema
			value = gkrb_econ_current_power
		}
		multiply_variable = {
			which = gkrb_temp_ema
			value = value:gkrb_econ_ema_alpha
		}
		# score = score + temp
		change_variable = {
			which = gkrb_econ_score
			value = gkrb_temp_ema
		}
	}
}

# Helper to add a single resource's weighted value to the score
# Now uses resource_revenue_compare directly (no caching needed)
gkrb_add_weighted_resource_score = {
	export_trigger_value_to_variable = {
		trigger = resource_revenue_compare
		parameters = {
			resource = $RESOURCE$
		}
		variable = gkrb_temp_score
	}
	multiply_variable = {
		which = gkrb_temp_score
		value = event_target:global_event_country.gkrb_econ_weight_$RESOURCE$
	}
	change_variable = {
		which = gkrb_econ_current_power
		value = gkrb_temp_score
	}
}
