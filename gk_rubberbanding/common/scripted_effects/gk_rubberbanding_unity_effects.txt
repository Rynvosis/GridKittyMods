####################################
# Unity Score Calculations         #
####################################

# Calculate unity income (net monthly gain from production)
gkrb_calculate_unity_income = {
	export_trigger_value_to_variable = {
		trigger = resource_income_compare
		parameters = {
			resource = unity
		}
		variable = gkrb_unity_income
	}
}

# Calculate unity stockpile delta (catches event-based gains)
# Also saves current stockpile as previous for next month
gkrb_calculate_unity_stockpile_delta = {
	# Get current stockpile
	export_resource_stockpile_to_variable = {
		resource = unity
		variable = gkrb_unity_stockpile_current
	}
	
	# Initialize previous stockpile if not set
	if = {
		limit = { NOT = { is_variable_set = gkrb_unity_stockpile_previous } }
		set_variable = {
			which = gkrb_unity_stockpile_previous
			value = gkrb_unity_stockpile_current
		}
	}
	
	# Calculate delta (current - previous)
	set_variable = {
		which = gkrb_unity_stockpile_delta
		value = gkrb_unity_stockpile_current
	}
	subtract_variable = {
		which = gkrb_unity_stockpile_delta
		value = gkrb_unity_stockpile_previous
	}

	# Save current as previous for next month
	set_variable = {
		which = gkrb_unity_stockpile_previous
		value = gkrb_unity_stockpile_current
	}
}

# Determine raw unity gain (max of income or stockpile delta)
gkrb_calculate_unity_raw_gain = {
	# Use whichever is greater
	if = {
		limit = {
			check_variable = {
				which = gkrb_unity_stockpile_delta
				value > gkrb_unity_income
			}
		}
		set_variable = {
			which = gkrb_unity_raw_gain
			value = gkrb_unity_stockpile_delta
		}
	}
	else = {
		set_variable = {
			which = gkrb_unity_raw_gain
			value = gkrb_unity_income
		}
	}
	
	# Clamp to 0 minimum (ignore negative from spending traditions)
	if = {
		limit = {
			check_variable = {
				which = gkrb_unity_raw_gain
				value < 0
			}
		}
		set_variable = {
			which = gkrb_unity_raw_gain
			value = 0
		}
	}

}

# Add effective unity to running total using bucket system
gkrb_accumulate_unity_score = {
	# Calculate effective unity (raw / tradition_cost_mult)
	set_variable = {
		which = gkrb_unity_effective
		value = value:gkrb_unity_effective_gain
	}
	
	# Initialize buckets if not set
	if = {
		limit = { NOT = { is_variable_set = gkrb_unity_total_low } }
		set_variable = { which = gkrb_unity_total_low value = 0 }
		set_variable = { which = gkrb_unity_total_high value = 0 }
		set_variable = { which = gkrb_unity_months_elapsed value = 0 }
	}
	
	# Add effective unity to low bucket
	change_variable = {
		which = gkrb_unity_total_low
		value = gkrb_unity_effective
	}
	
	# Increment months elapsed
	change_variable = {
		which = gkrb_unity_months_elapsed
		value = 1
	}
	
	# Check for overflow: if low >= threshold, transfer to high
	while = {
		limit = {
			check_variable = {
				which = gkrb_unity_total_low
				value >= @gkrb_unity_bucket_threshold
			}
		}
		subtract_variable = {
			which = gkrb_unity_total_low
			value = @gkrb_unity_bucket_threshold
		}
		change_variable = {
			which = gkrb_unity_total_high
			value = 1
		}
	}
	
	# Cache the normalized score for comparisons (total / months)
	set_variable = {
		which = gkrb_unity_score
		value = value:gkrb_unity_score
	}

	# Scratch values no longer needed post-accumulation
	clear_variable = gkrb_unity_effective
	clear_variable = gkrb_unity_raw_gain
	clear_variable = gkrb_unity_stockpile_delta
}

# Main entry point: calculate unity score for this month
# Calls all sub-effects in order
gkrb_calculate_unity_score = {
	gkrb_calculate_unity_income = yes
	gkrb_calculate_unity_stockpile_delta = yes
	gkrb_calculate_unity_raw_gain = yes
	gkrb_accumulate_unity_score = yes
}

# Clear all unity rubberbanding modifiers
gkrb_clear_unity_rubberbanding_modifiers = {
	remove_modifier = gkrb_unity_unity_add
	remove_modifier = gkrb_unity_mod
	remove_modifier = gkrb_unity_behind_mod
}

# Apply unity rubberbanding when ahead (upkeep penalty)
gkrb_apply_ahead_unity_rubberbanding_modifiers = {
	# Apply marker modifier with upkeep (tooltip)
	add_modifier = {
		modifier = gkrb_unity_unity_add
		days = -1
		multiplier = gkrb_unity_unity_adjust
	}
	add_modifier = {
		modifier = gkrb_unity_mod
		days = -1
		multiplier = gkrb_unity_mult
	}
}

# Apply unity rubberbanding when behind (catch-up bonus)
gkrb_apply_behind_unity_rubberbanding_modifiers = {
	add_modifier = {
		modifier = gkrb_unity_behind_mod
		days = -1
		multiplier = gkrb_unity_mult
	}
}
