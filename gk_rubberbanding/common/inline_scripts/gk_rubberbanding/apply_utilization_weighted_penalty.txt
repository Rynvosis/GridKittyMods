# Inline script: Apply utilization-weighted penalty for a resource
# Penalty_r = utilization_r × Mal_total × EMA_Revenue_r
# Parameters:
#   $RESOURCE$ = Resource name (energy, minerals, etc.)
#
# Requires: 
#   - gkrb_econ_utilization_ratio_$RESOURCE$ to be set (from EMA use/revenue)
#   - gkrb_econ_ema_revenue_$RESOURCE$ to be set
#   - gkrb_total_penalty to be set (= Mal_total = clamp(AR, 0, 1)²)
# Applies modifier: gkrb_econ_$RESOURCE$_mod

# Calculate penalty using script value (utilization × penalty_mult × ema_revenue)
set_variable = {
	which = gkrb_econ_penalty_$RESOURCE$
	value = value:gkrb_resource_penalty_amount|RESOURCE|$RESOURCE$|
}

# Round to 2 decimal places
round_variable_to_closest = {
	which = gkrb_econ_penalty_$RESOURCE$
	value = 0.01
}

# Only apply if penalty is significant (>= 0.1)
if = {
	limit = {
		check_variable = {
			which = gkrb_econ_penalty_$RESOURCE$
			value >= 0.1
		}
	}
	add_modifier = {
		modifier = gkrb_econ_$RESOURCE$_mod
		days = -1
		multiplier = gkrb_econ_penalty_$RESOURCE$
	}
}
