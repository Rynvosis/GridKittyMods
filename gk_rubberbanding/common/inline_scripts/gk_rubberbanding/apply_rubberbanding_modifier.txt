# Inline script for applying rubberbanding modifier
# Parameters:
#   $RESOURCE$ - Resource name (e.g., energy)
#   $TYPE$     - System type (tech or econ)
#
# Uses resource_revenue_compare which gives us the final revenue directly.
# Upkeep = revenue * rb_mult (if 100% ahead, upkeep = 100% of revenue)

remove_modifier = gkrb_$TYPE$_$RESOURCE$_mod

# Get current revenue directly - no mult math needed
export_trigger_value_to_variable = {
	trigger = resource_revenue_compare
	parameters = {
		resource = $RESOURCE$
	}
	variable = gkrb_$TYPE$_$RESOURCE$_adjust
}

# Upkeep = revenue * rubberbanding_mult
multiply_variable = {
	which = gkrb_$TYPE$_$RESOURCE$_adjust
	value = gkrb_$TYPE$_mult
}

# Round to 2 decimal places
round_variable_to_closest = {
	which = gkrb_$TYPE$_$RESOURCE$_adjust
	value = 0.01
}

# Apply modifier (upkeep scales with how far ahead you are)
add_modifier = {
	modifier = gkrb_$TYPE$_$RESOURCE$_mod
	days = -1
	multiplier = gkrb_$TYPE$_$RESOURCE$_adjust
}
