# Calculate EMA revenue for abstract resources (research/unity)
# These don't track utilization - lead factors replace that concept

# Get current revenue
set_variable = {
	which = gkrb_effective_revenue_$RESOURCE$
	value = value:gkrb_effective_resource_revenue|RESOURCE|$RESOURCE$|
}

# Update EMA revenue
create_variable_if_not_exists = {
	VARIABLE = gkrb_econ_ema_revenue_$RESOURCE$
	VALUE_IF_CREATED = gkrb_effective_revenue_$RESOURCE$
}
set_variable = {
	which = gkrb_econ_ema_revenue_$RESOURCE$
	value = value:gkrb_ema_update_revenue|RESOURCE|$RESOURCE$|
}
