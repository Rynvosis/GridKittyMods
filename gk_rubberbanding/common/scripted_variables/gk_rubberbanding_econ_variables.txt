####################################
# Value-Weighted Econ Rubberbanding #
####################################
# This system calculates penalties based on CONSUMPTION not production.
# EconValue_r = max(0, Use_r - k×signedSqrt(maintenance_r)) × weight_r
# Where:
#   Use_r = Rev_r - ΔStock_r (actual consumption, not overproduction)
#   maintenance_r = sum of approved upkeep categories (pops, jobs, buildings)
#   signedSqrt = sign(x) × sqrt(abs(x)) (diminishing returns on maintenance)
#   weight_r = Stellaris standard resource weights

# EMA (Exponential Moving Average) parameters for smoothing
# Half-life in months - after this many months, old data contributes 50%
@gkrb_econ_ema_halflife_months = 40

# maintenance discount factor (k)
# Higher = more discount for productive spending
# At k=0.8: spending 100 on maintenance discounts ~8 effective use
# At k=1.0: spending 100 on maintenance discounts 10 effective use
@gkrb_maintenance_discount = 0.8

# Bloc power inheritance multipliers
# These determine what fraction of the pairwise power bonus applies
@gkrb_bloc_mult_federation = 1.0
@gkrb_bloc_mult_defensive_pact = 0.5
@gkrb_bloc_mult_overlord = 0.25
@gkrb_bloc_mult_vassal = 1.0

####################################
