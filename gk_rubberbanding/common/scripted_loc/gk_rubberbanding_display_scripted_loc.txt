defined_text = {
	name = GKRBEnergyRow
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_penalty_energy
				value >= 0.1
			}
		}
		localization_key = "GKRB_ROW_ENERGY_UPKEEP"
	}
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_penalty_energy
				value < 0.1
			}
		}
		localization_key = "GKRB_EMPTY"
	}
}
