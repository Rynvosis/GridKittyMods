defined_text = {
	name = GKRBEnergyRow
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_energy_adjust
				value >= 0.1
			}
		}
		localization_key = "GKRB_ROW_ENERGY_POS"
	}
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_energy_adjust
				value <= -0.1
			}
		}
		localization_key = "GKRB_ROW_ENERGY_NEG"
	}
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_energy_adjust
				value < 0.1
			}
			check_variable = {
				which = gkrb_econ_energy_adjust
				value > -0.1
			}
		}
		localization_key = "GKRB_EMPTY"
	}
}
