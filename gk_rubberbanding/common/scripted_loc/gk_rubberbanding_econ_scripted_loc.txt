# Bloc power display (show only if positive)
defined_text = {
	name = GKRBBlocPowerIncrease
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_bloc_power_increase
				value > 0
			}
		}
		localization_key = GKRB_BLOC_POWER_POSITIVE
	}
	text = {
		trigger = {
			check_variable = {
				which = gkrb_econ_bloc_power_increase
				value = 0
			}
		}
		localization_key = GKRB_EMPTY
	}
}
