# Overwrite of vanilla ship_components/bio_lm_ship_armor.txt.
# Vanilla uses flat food upkeep (@$SIZE$_t$TIER$_upkeep_energy). For lithogenesis
# empires (graphical_culture = lithogenesis_01) we swap food -> minerals per the
# canonical paired-trigger pattern used elsewhere in this mod. Alloys upkeep,
# living-metal cost, modifiers, and ship_modifier are preserved verbatim.
modifier = {
	ship_hull_add = @bio_ship_hull_$SIZE$_$TIER$
	ship_armor_add = @bio_ship_armor_$SIZE$_$TIER$
	ship_armor_regen_add_static = @bio_lm_ship_armor_regen_$SIZE$_$TIER$
}

ship_modifier = {
	weapon_type_energy_weapon_fire_rate_mult = @inf_energy_speed
}

resources = {
	category = ship_components
	cost = {
		sr_living_metal = @armor_$SIZE$_t$TIER$_cost
	}
	upkeep = {
		food = @$SIZE$_t$TIER$_upkeep_energy
		trigger = {
			from = {
				country_uses_bio_ships = yes
				NOT = { graphical_culture = lithogenesis_01 }
			}
		}
	}
	upkeep = {
		minerals = @$SIZE$_t$TIER$_upkeep_energy
		mult = 0.8
		trigger = {
			from = {
				country_uses_bio_ships = yes
				graphical_culture = lithogenesis_01
			}
		}
	}
	upkeep = {
		alloys = @$SIZE$_t$TIER$_upkeep_alloys
	}
}
