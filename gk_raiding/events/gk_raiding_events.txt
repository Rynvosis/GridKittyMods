@gk_pillage_progress_threshold = 10
@gk_decay_rate_at_peace = 0.90
@gk_decay_rate_at_war = 0.95
@gk_decay_clear_threshold = 1

# Machine Despoilers Setup #
############################
# This = country
# Converts 400 starting machine pops to organic battery pops
# Sets secondary species to grid amalgamation slavery
# Ensures a farming district on capital
country_event = {
	id = gk_raiding.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_valid_civic = civic_gk_barbaric_despoilers_machine
	}

	immediate = {
		# Set secondary species to grid amalgamation slavery
		every_owned_species = {
			limit = {
				is_robotic = no
				NOT = { is_same_species = root.owner_species }
			}
			save_event_target_as = gk_battery_species
			set_citizenship_type = {
				country = root
				type = citizenship_slavery
				cooldown = no
			}
			set_slavery_type = {
				country = root
				type = slavery_matrix
				cooldown = no
			}
		}

		capital_scope = {
			# Kill 400 machine pops from the main species pop group
			random_owned_pop_group = {
				limit = {
					is_same_species = root.owner_species
				}
				kill_pop_group = {
					pop_group = this
					amount = 400
				}
			}

			# Create 400 organic battery pops
			create_pop_group = {
				species = event_target:gk_battery_species
				size = 400
			}

			# Ensure at least 1 farming district, trading a generator for it if possible
			if = {
				limit = {
					NOT = {
						has_district = district_farming
					}
				}
				add_district = district_farming
			}

			if = {
				limit = {
					has_district = district_generator
				}
				remove_district = district_generator
			}
		}
	}
}

# Daily bombardment - pillage progress
# This = Planet
# From = Bombarding country
planet_event = {
	id = gk_raiding.100
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_orbital_bombardment = yes
		has_orbital_bombardment_stance = raiding
		exists = owner
		from = {
			gk_can_raid_resources = yes
		}
		# Can still be pillaged
		gk_can_be_pillaged = yes
	}
	immediate = {
		from = {
			save_event_target_as = gk_raider
		}
		# Initialize
		create_variable_if_not_exists = {
			VARIABLE = gk_pillage_tier
			VALUE_IF_CREATED = 0
		}
		create_variable_if_not_exists = {
			VARIABLE = gk_pillage_progress
			VALUE_IF_CREATED = 0
		}
		# increased chance to pillage based on devastation.
		# based on power curve.
		random_list = {
			10 = {
				modifier = {
					add = value:gk_devastation_curve
				}
				change_variable = {
					which = gk_pillage_progress
					value = 1
				}
			}
			110 = {
				modifier = {
					subtract = value:gk_devastation_curve
				}
			}
		}
		# Check for tier advancement (progress >= 20)
		if = {
			limit = {
				check_variable = {
					which = gk_pillage_progress
					value >= @gk_pillage_progress_threshold
				}
			}
			# Reset progress
			set_variable = {
				which = gk_pillage_progress
				value = 0
			}
			# Apply pillage with immediate resources + notification
			gk_pillage_on_bombardment = yes
		}
	}
}

# Yearly decay of raidee plundered/emancipated counters
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding.50
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			is_variable_set = gk_raidee_plundered
			is_variable_set = gk_raidee_emancipated
		}
	}
	immediate = {
		# Decay plundered by 10% (5% if defending against raider)
		if = {
			limit = {
				is_variable_set = gk_raidee_plundered
			}
			if = {
				limit = {
					any_war = {
						any_defender = {
							is_same_value = root
						}
						OR = {
							using_war_goal = {
								type = wg_gk_emancipation
								owner = attacker
							}
							using_war_goal = {
								type = wg_plunder
								owner = attacker
							}
							using_war_goal = {
								type = wg_gk_robo_liberation
								owner = attacker
							}
							using_war_goal = {
								type = wg_gk_reaping
								owner = attacker
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_peace
				}
			}
			# Clear if below threshold and not at war
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_plundered
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_plundered
			}
		}
		# Decay emancipated by 10% (5% if defending against raider)
		if = {
			limit = {
				is_variable_set = gk_raidee_emancipated
			}
			if = {
				limit = {
					any_war = {
						any_defender = {
							is_same_value = root
						}
						OR = {
							using_war_goal = {
								type = wg_gk_emancipation
								owner = attacker
							}
							using_war_goal = {
								type = wg_plunder
								owner = attacker
							}
							using_war_goal = {
								type = wg_gk_robo_liberation
								owner = attacker
							}
							using_war_goal = {
								type = wg_gk_reaping
								owner = attacker
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_peace
				}
			}
			# Clear if below threshold and not at war
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_emancipated
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_emancipated
			}
		}
		# Apply/update recovery modifier based on plunder percentage
		gk_update_plunder_recovery_modifier = yes
	}
}

# Yearly cleanup of raid tracker leaders and cap flags
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding.52
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		# Only run if we have any raid cap flags
		any_country = {
			has_country_flag = gk_raid_cap_reached_by_@root
		}
	}
	immediate = {
		# Check each country that has a cap flag for us
		every_country = {
			limit = {
				has_country_flag = gk_raid_cap_reached_by_@root
			}
			# If no longer at war with this raider, clean up
			if = {
				limit = {
					NOT = {
						root = {
							any_war = {
								is_war_participant = {
									who = prevprev
								}
							}
						}
					}
				}
				remove_country_flag = gk_raid_cap_reached_by_@root
			}
		}
		# Clean up tracker leaders for pairs no longer at war
		save_event_target_as = gk_raider
		event_target:global_event_country = {
			every_owned_leader = {
				limit = {
					has_leader_flag = gk_raid_tracker
					has_leader_flag = gk_raid_tracker_attacker_@root
				}
				# Check if still at war with the defender
				# We need to find the defender from the flag
				# Since we can't iterate flags, just check if raider is at war at all
				# If raider has no wars, clean up all their trackers
				if = {
					limit = {
						root = { is_at_war = no }
					}
					kill_leader = { show_notification = no }
				}
			}
		}
	}
}

# Reaping surrender - display total forfeited pops (mass abduction)
# This = Winner (raider)
# From = Loser (raidee) - passed via scopes
# Abduction already happened, just display count from receipt
country_event = {
	id = gk_raiding.700
	title = gk_raiding.700.title
	desc = gk_raiding.700.desc
	picture = GFX_evt_burning_city
	show_sound = event_planetary_riot
	is_triggered_only = yes
	immediate = {
		# Save event targets
		save_event_target_as = gk_raider
		from = {
			save_event_target_as = gk_raidee
			save_event_target_as = gk_raiding_target
		}
		# Retrieve the loot receipt to display values
		gk_retrieve_loot_receipt = yes
		# Copy values to root for localisation
		gk_copy_receipt_values_for_display = yes
	}
	option = {
		name = gk_raiding.700.accept
		custom_tooltip = gk_raiding.700.tooltip
		hidden_effect = {
			# Apply per-pop opinion penalty to all countries with communications
			every_country = {
				limit = {
					has_communications = root
					NOT = { is_country = root }
					OR = {
						is_country_type = default
						has_ai_personality = awakened_fallen_empire_xenophile
					}
				}
				while = {
					count = root.gk_forfeited_pops
					add_opinion_modifier = { who = root modifier = opinion_gk_reaped_pops }
				}
			}
			gk_destroy_loot_receipt = yes
		}
	}
}

##############################
# Applies opinion penalty when a Reaping war is declared
# This = Country (every country in the war, via on_war_beginning)
# From = War
country_event = {
	id = gk_raiding.800
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		# Only fire for the attacking war leader using Reaping
		from = {
			using_war_goal = {
				type = wg_gk_reaping
				owner = root
			}
			any_attacker = {
				is_war_leader = yes
				is_same_value = root
			}
		}
	}
	immediate = {
		# Apply flat opinion penalty to all countries with communications
		every_country = {
			limit = {
				has_communications = root
				NOT = { is_country = root }
				OR = {
					is_country_type = default
					has_ai_personality = awakened_fallen_empire_xenophile
				}
			}
			add_opinion_modifier = { who = root modifier = opinion_gk_declared_reaping }
		}
	}
}
