# Machine Despoilers Setup #
############################
# This = country
# Converts 400 starting machine pops to organic battery pops
# Sets secondary species to grid amalgamation slavery
# Ensures a farming district on capital
country_event = {
	id = gk_raiding.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_valid_civic = civic_gk_barbaric_despoilers_machine
	}

	immediate = {
		# Set secondary species to grid amalgamation slavery
		every_owned_species = {
			limit = {
				is_robotic = no
				NOT = { is_same_species = root.owner_species }
			}
			save_event_target_as = gk_battery_species
			set_citizenship_type = {
				country = root
				type = citizenship_slavery
				cooldown = no
			}
			set_slavery_type = {
				country = root
				type = slavery_matrix
				cooldown = no
			}
		}

		capital_scope = {
			# Kill 400 machine pops from the main species pop group
			random_owned_pop_group = {
				limit = {
					is_same_species = root.owner_species
				}
				kill_pop_group = {
					pop_group = this
					amount = 400
				}
			}

			# Create 400 organic battery pops
			create_pop_group = {
				species = event_target:gk_battery_species
				size = 400
			}

			# Ensure at least 1 farming district, trading a generator for it if possible
			if = {
				limit = {
					NOT = {
						has_district = district_farming
					}
				}
				add_district = district_farming
			}

			if = {
				limit = {
					has_district = district_generator
				}
				remove_district = district_generator
			}
		}
	}
}

# Daily devastation from bombardment or ground combat - immediate pillage packets
# Also resolves forced orbital surrender before ownership flips.
# This = Planet
# From = Damaging country
planet_event = {
	id = gk_raiding.100
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		exists = owner
		exists = from
		OR = {
			AND = {
				from = {
					gk_can_raid_resources = yes
				}
				gk_can_be_pillaged = yes
			}
			AND = {
				from = {
					gk_can_pop_raid_personally = yes
				}
				owner = {
					from = {
						OR = {
							prev = { gk_has_standard_raiding_war_with = yes }
							prev = { gk_has_emancipation_war_with = yes }
							prev = { gk_has_robo_liberation_war_with = yes }
							prev = { gk_has_reaping_war_with = yes }
						}
					}
				}
			}
		}
	}
	immediate = {
		from = {
			save_event_target_as = gk_raider
		}
		if = {
			limit = {
				from = {
					gk_can_raid_resources = yes
				}
				gk_can_be_pillaged = yes
			}
			gk_pillage_on_devastation = yes
		}
		if = {
			limit = {
				has_orbital_bombardment = yes
				num_armies < 1
				NOT = { has_ground_combat = yes }
				NOT = { has_planet_flag = gk_pillage_forced_surrender_resolved_by_@from }
				OR = {
					AND = {
						from = {
							gk_can_pillage = yes
							is_at_war_with = prev.owner
						}
					}
					AND = {
						from = {
							gk_can_pop_raid_personally = yes
						}
						owner = {
							from = {
								OR = {
									prev = { gk_has_standard_raiding_war_with = yes }
									prev = { gk_has_emancipation_war_with = yes }
									prev = { gk_has_robo_liberation_war_with = yes }
									prev = { gk_has_reaping_war_with = yes }
								}
							}
						}
					}
				}
			}
			set_planet_flag = gk_pillage_forced_surrender_resolved_by_@from
			from = {
				country_event = {
					id = gk_raiding.200
					scopes = {
						from = root.owner
						fromfrom = root
					}
				}
			}
		}
	}
}

# Yearly decay of raidee plundered/emancipated counters
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding.50
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			is_variable_set = gk_raidee_plundered
			is_variable_set = gk_raidee_emancipated
		}
	}
	immediate = {
		# Decay plundered by 10% (5% if defending against raider)
		if = {
			limit = {
				is_variable_set = gk_raidee_plundered
			}
			if = {
				limit = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = root side = defenders }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
									using_war_goal = { type = wg_gk_reaping owner = attacker }
								}
							}
							AND = {
								is_war_participant = { who = root side = attackers }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
									using_war_goal = { type = wg_gk_reaping owner = defender }
								}
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_peace
				}
			}
			# Clear if below threshold and not at war
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_plundered
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_plundered
			}
		}
		# Decay emancipated by 10% (5% if defending against raider)
		if = {
			limit = {
				is_variable_set = gk_raidee_emancipated
			}
			if = {
				limit = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = root side = defenders }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
									using_war_goal = { type = wg_gk_reaping owner = attacker }
								}
							}
							AND = {
								is_war_participant = { who = root side = attackers }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
									using_war_goal = { type = wg_gk_reaping owner = defender }
								}
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_peace
				}
			}
			# Clear if below threshold and not at war
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_emancipated
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_emancipated
			}
		}
		# Apply/update recovery modifier based on plunder percentage
		gk_update_plunder_recovery_modifier = yes
	}
}

# Yearly cleanup of raid tracker leaders and cap flags
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding.52
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		OR = {
			# Only run if we have any raid cap flags
			any_country = {
				has_country_flag = gk_raid_cap_reached_by_@root
			}
			# Or any invasion cashout markers keyed to a country that may no longer be at war with us
			any_country = {
				root = {
					has_country_flag = gk_has_pillage_invasion_cashed_out_by_@prev
				}
			}
		}
	}
	immediate = {
		# Check each country that has a cap flag for us
		every_country = {
			limit = {
				has_country_flag = gk_raid_cap_reached_by_@root
			}
			# If no longer at war with this raider, clean up
			if = {
				limit = {
					NOT = {
						root = {
							any_war = {
								is_war_participant = {
									who = prevprev
								}
							}
						}
					}
				}
				remove_country_flag = gk_raid_cap_reached_by_@root
			}
		}
		# Clean up tracker leaders for pairs no longer at war
		save_event_target_as = gk_raider
		event_target:global_event_country = {
			every_owned_leader = {
				limit = {
					has_leader_flag = gk_raid_tracker
					has_leader_flag = gk_raid_tracker_attacker_@root
				}
				# Check if still at war with the defender
				# We need to find the defender from the flag
				# Since we can't iterate flags, just check if raider is at war at all
				# If raider has no wars, clean up all their trackers
				if = {
					limit = {
						root = { is_at_war = no }
					}
					kill_leader = { show_notification = no }
				}
			}
		}
		# Clean up invasion cashout flags for raiders we are no longer at war with
		every_country = {
			limit = {
				root = {
					has_country_flag = gk_has_pillage_invasion_cashed_out_by_@prev
				}
			}
			if = {
				limit = {
					root = {
						NOT = {
							any_war = {
								is_war_participant = { who = prev }
							}
						}
					}
				}
				save_event_target_as = gk_raider
				root = {
					gk_clear_invasion_pillage_cashed_out_flags = yes
				}
			}
		}
	}
}
