# From = Bombarder country
# FromFrom = Biggest fleet bombarding
# local_kill_pop_amount = exact number of pops transferred

	id = gk_raiding_pops.100
	hide_window = yes
	is_triggered_only = yes

		has_orbital_bombardment = yes

# Yearly decay of raidee plundered/emancipated counters
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding_pops.110
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			is_variable_set = gk_raidee_plundered
			is_variable_set = gk_raidee_emancipated
		}
	}

	immediate = {
		if = {
			limit = {
				is_variable_set = gk_raidee_plundered
			}
			if = {
				limit = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = root side = defenders }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
									using_war_goal = { type = wg_gk_reaping owner = attacker }
								}
							}
							AND = {
								is_war_participant = { who = root side = attackers }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
									using_war_goal = { type = wg_gk_reaping owner = defender }
								}
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_peace
				}
			}
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_plundered
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_plundered
			}
		}
		if = {
			limit = {
				is_variable_set = gk_raidee_emancipated
			}
			if = {
				limit = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = root side = defenders }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
									using_war_goal = { type = wg_gk_reaping owner = attacker }
								}
							}
							AND = {
								is_war_participant = { who = root side = attackers }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
									using_war_goal = { type = wg_gk_reaping owner = defender }
								}
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_peace
				}
			}
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_emancipated
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_emancipated
			}
		}
	}
}

# Yearly cleanup of raid tracker leaders and cap flags
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding_pops.120
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_country = {
			has_country_flag = gk_raid_cap_reached_by_@root
		}
	}

	immediate = {
		every_country = {
			limit = {
				has_country_flag = gk_raid_cap_reached_by_@root
			}
			if = {
				limit = {
					NOT = {
						root = {
							any_war = {
								is_war_participant = {
									who = prevprev
								}
							}
						}
					}
				}
				remove_country_flag = gk_raid_cap_reached_by_@root
			}
		}
		save_event_target_as = gk_raider
		event_target:global_event_country = {
			every_owned_leader = {
				limit = {
					has_leader_flag = gk_raid_tracker
					has_leader_flag = gk_raid_tracker_attacker_@root
				}
				if = {
					limit = {
						root = { is_at_war = no }
					}
					kill_leader = { show_notification = no }
				}
			}
		}
	}
}

# On planet attackers win - instant pop raid
# This = Attacking Country
# From = Defending Country
# FromFrom = Planet
country_event = {
	id = gk_raiding_pops.210
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		fromfrom = {
			is_colony = yes
		}
		gk_can_pop_raid_personally = yes
		from = {
			OR = {
				gk_has_standard_raiding_war_with = yes
				gk_has_emancipation_war_with = yes
				gk_has_robo_liberation_war_with = yes
				gk_has_reaping_war_with = yes
			}
		}
	}

	immediate = {
		save_event_target_as = gk_raider
		from = {
			save_event_target_as = gk_raidee
		}
		fromfrom = {
			save_event_target_as = gk_target_planet
			save_event_target_as = gk_raiding_target
		}

		gk_abduct_on_invasion = yes
		gk_retrieve_pop_result_tracker = yes
		gk_copy_pop_result_values_for_display = yes

		if = {
			limit = {
				check_variable = { which = gk_forfeited_pops value > 0 }
			}
			event_target:gk_raider = {
				create_message = {
					type = MESSAGE_TYPE_GK_POP_RAIDED
					localization = MESSAGE_GK_POP_RAIDED_INVASION
					days = 30
					target = event_target:gk_target_planet
					variable = {
						type = name
						localization = PLANET
						scope = event_target:gk_target_planet
					}
					variable = {
						type = variable
						localization = POPS
						varname = gk_forfeited_pops
						scope = event_target:gk_raider
					}
				}
			}
			event_target:gk_raidee = {
				create_message = {
					type = MESSAGE_TYPE_GK_OUR_POPS_RAIDED
					localization = MESSAGE_GK_OUR_POPS_RAIDED_INVASION
					days = 30
					target = event_target:gk_target_planet
					variable = {
						type = name
						localization = PLANET
						scope = event_target:gk_target_planet
					}
					variable = {
						type = name
						localization = FROM
						scope = event_target:gk_raider
					}
					variable = {
						type = variable
						localization = POPS
						varname = gk_forfeited_pops
						scope = event_target:gk_raider
					}
				}
			}
		}

		gk_destroy_pop_result_tracker = yes
	}
}

# Bucket completed — offer escalation to side leader
# This = gk_cap_raider (side leader)
# From = raidee country
country_event = {
	id = gk_raiding_pops.500
	title = GK_RAIDING_ESCALATION_TITLE
	desc = {
		trigger = {
			event_target:gk_raid_tracker = {
				check_variable = { which = gk_escalation_tier value >= 4 }
			}
		}
		text = GK_RAIDING_ESCALATION_DESC_FINAL
	}
	desc = {
		trigger = { always = yes }
		text = GK_RAIDING_ESCALATION_DESC
	}
	picture = GFX_evt_ground_combat
	show_sound = event_alien_nature
	is_triggered_only = yes

	trigger = {
		# Only show if war still ongoing and cap still reached
		is_at_war_with = from
		from = {
			has_country_flag = gk_raid_cap_reached_by_@root
		}
	}

	immediate = {
		save_event_target_as = gk_cap_raider
		from = {
			save_event_target_as = gk_raidee
		}
		gk_get_or_create_raid_tracker = yes
		# Compute next bucket yield for display: raidee pops * next tier rate
		set_variable = { which = gk_display_bucket_remaining value = from.trigger:count_owned_pop_amount }
		# Next tier rate lookup (current tier just completed, so next = tier + 1)
		set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_1 }
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 1 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_2 }
		}
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 2 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_3 }
		}
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 3 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_4 }
		}
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 4 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_5 }
		}
		multiply_variable = {
			which = gk_display_bucket_remaining
			value = gk_display_next_rate
		}
		clear_variable = gk_display_next_rate
	}

	# Option A: Consolidate
	option = {
		name = GK_RAIDING_ESCALATION_CONSOLIDATE
		custom_tooltip = GK_RAIDING_ESCALATION_CONSOLIDATE_TOOLTIP
		hidden_effect = {
			# Re-fire after 360 days
			country_event = {
				id = gk_raiding_pops.500
				days = 360
				scopes = { from = event_target:gk_raidee }
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 3
				OR = {
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
				}
			}
			modifier = {
				factor = 2
				resource_stockpile_compare = {
					resource = influence
					value < 100
				}
			}
		}
	}

	# Option B: Press the Raid
	option = {
		name = {
			trigger = {
				event_target:gk_raid_tracker = {
					check_variable = { which = gk_escalation_tier value >= 4 }
				}
			}
			text = GK_RAIDING_ESCALATION_PRESS_FINAL
		}
		name = GK_RAIDING_ESCALATION_PRESS
		custom_tooltip = GK_RAIDING_ESCALATION_PRESS_TOOLTIP
		allow = {
			resource_stockpile_compare = {
				resource = influence
				value >= value:gk_escalation_influence_cost
			}
		}
		hidden_effect = {
			# Charge influence
			add_resource = {
				influence = -1
				mult = value:gk_escalation_influence_cost
			}
			# Advance bucket (rate is derived from tier via value:gk_bucket_rate_for_tier)
			event_target:gk_raid_tracker = {
				change_variable = { which = gk_escalation_tier value = 1 }
				set_variable = { which = gk_bucket_pops_taken value = 0 }
			}
			# Remove cap flag so raiding can continue
			from = {
				remove_country_flag = gk_raid_cap_reached_by_@root
			}
		}
		ai_chance = {
			factor = 50
			modifier = {
				factor = 3
				OR = {
					has_ethic = ethic_militarist
					has_ethic = ethic_fanatic_militarist
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
				}
			}
			modifier = {
				factor = 0.5
				resource_stockpile_compare = {
					resource = influence
					value < 100
				}
			}
		}
	}

	after = {
		hidden_effect = {
			clear_variable = gk_display_bucket_remaining
		}
	}
}
