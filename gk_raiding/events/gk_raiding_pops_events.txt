# Pop abducted by raiding bombardment stance
# This = Planet where pop was "killed" (source)
# From = Bombarder country
# FromFrom = Biggest fleet bombarding
# local_kill_pop_amount = exact number of pops transferred
planet_event = {
	id = gk_raiding_pops.100
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_orbital_bombardment = yes
		has_orbital_bombardment_stance = raiding
	}

	immediate = {
		from = {
			save_event_target_as = gk_raider
		}
		owner = {
			gk_track_raided_pops = { AMOUNT = root.local_kill_pop_amount }
		}
	}
}

# Yearly decay of raidee plundered/emancipated counters
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding_pops.110
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			is_variable_set = gk_raidee_plundered
			is_variable_set = gk_raidee_emancipated
		}
	}

	immediate = {
		if = {
			limit = {
				is_variable_set = gk_raidee_plundered
			}
			if = {
				limit = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = root side = defenders }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
									using_war_goal = { type = wg_gk_reaping owner = attacker }
								}
							}
							AND = {
								is_war_participant = { who = root side = attackers }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
									using_war_goal = { type = wg_gk_reaping owner = defender }
								}
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_plundered
					value = @gk_decay_rate_at_peace
				}
			}
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_plundered
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_plundered
			}
		}
		if = {
			limit = {
				is_variable_set = gk_raidee_emancipated
			}
			if = {
				limit = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = root side = defenders }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
									using_war_goal = { type = wg_gk_reaping owner = attacker }
								}
							}
							AND = {
								is_war_participant = { who = root side = attackers }
								OR = {
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
									using_war_goal = { type = wg_gk_reaping owner = defender }
								}
							}
						}
					}
				}
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_war
				}
			}
			else = {
				multiply_variable = {
					which = gk_raidee_emancipated
					value = @gk_decay_rate_at_peace
				}
			}
			if = {
				limit = {
					is_at_war = no
					check_variable = {
						which = gk_raidee_emancipated
						value <= @gk_decay_clear_threshold
					}
				}
				clear_variable = gk_raidee_emancipated
			}
		}
	}
}

# Yearly cleanup of raid tracker leaders and cap flags
# This = Country scope (from on_yearly_pulse_country)
country_event = {
	id = gk_raiding_pops.120
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_country = {
			OR = {
				has_country_flag = gk_raid_cap_reached_by_@root
				has_country_flag = gk_pop_raiding_terminated_by_@root
				root = { has_country_flag = gk_pivoted_to_pillaging_against_@prev }
			}
		}
	}

	immediate = {
		every_country = {
			limit = {
				OR = {
					has_country_flag = gk_raid_cap_reached_by_@root
					has_country_flag = gk_pop_raiding_terminated_by_@root
					has_country_flag = gk_tribute_offer_pending_from_@root
					has_country_flag = gk_reforms_offer_pending_from_@root
					has_country_flag = gk_autonomy_offer_pending_from_@root
					root = { has_country_flag = gk_pivoted_to_pillaging_against_@prev }
				}
			}
			if = {
				limit = {
					NOT = {
						root = {
							any_war = {
								is_war_participant = {
									who = prevprev
								}
							}
						}
					}
				}
				remove_country_flag = gk_raid_cap_reached_by_@root
				remove_country_flag = gk_pop_raiding_terminated_by_@root
				remove_country_flag = gk_tribute_offer_pending_from_@root
				remove_country_flag = gk_reforms_offer_pending_from_@root
				remove_country_flag = gk_autonomy_offer_pending_from_@root
				root = { remove_country_flag = gk_pivoted_to_pillaging_against_@prev }
			}
		}
		save_event_target_as = gk_raider
		event_target:global_event_country = {
			every_owned_leader = {
				limit = {
					has_leader_flag = gk_raid_tracker
					has_leader_flag = gk_raid_tracker_attacker_@root
				}
				if = {
					limit = {
						root = { is_at_war = no }
					}
					kill_leader = { show_notification = no }
				}
			}
		}
	}
}

# On planet attackers win - instant pop raid
# This = Attacking Country
# From = Defending Country
# FromFrom = Planet
country_event = {
	id = gk_raiding_pops.210
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		fromfrom = {
			is_colony = yes
		}
		gk_can_pop_raid_personally = yes
		from = {
			OR = {
				gk_has_standard_raiding_war_with = yes
				gk_has_emancipation_war_with = yes
				gk_has_robo_liberation_war_with = yes
				gk_has_reaping_war_with = yes
			}
		}
	}

	immediate = {
		save_event_target_as = gk_raider
		from = {
			save_event_target_as = gk_raidee
		}
		fromfrom = {
			save_event_target_as = gk_target_planet
			save_event_target_as = gk_raiding_target
		}

		gk_abduct_on_invasion = yes
		gk_retrieve_pop_result_tracker = yes
		gk_copy_pop_result_values_for_display = yes

		if = {
			limit = {
				check_variable = { which = gk_forfeited_pops value > 0 }
			}
			event_target:gk_raider = {
				create_message = {
					type = MESSAGE_TYPE_GK_POP_RAIDED
					localization = MESSAGE_GK_POP_RAIDED_INVASION
					days = 30
					target = event_target:gk_target_planet
					variable = {
						type = name
						localization = PLANET
						scope = event_target:gk_target_planet
					}
					variable = {
						type = variable
						localization = POPS
						varname = gk_forfeited_pops
						scope = event_target:gk_raider
					}
				}
			}
			event_target:gk_raidee = {
				create_message = {
					type = MESSAGE_TYPE_GK_OUR_POPS_RAIDED
					localization = MESSAGE_GK_OUR_POPS_RAIDED_INVASION
					days = 30
					target = event_target:gk_target_planet
					variable = {
						type = name
						localization = PLANET
						scope = event_target:gk_target_planet
					}
					variable = {
						type = name
						localization = FROM
						scope = event_target:gk_raider
					}
					variable = {
						type = variable
						localization = POPS
						varname = gk_forfeited_pops
						scope = event_target:gk_raider
					}
				}
			}
		}

		# Check if we just depopulated the raidee entirely
		if = {
			limit = {
				event_target:gk_raidee = {
					count_owned_pop_amount = { count = 0 }
				}
			}
			country_event = {
				id = gk_raiding_pops.215
				scopes = { from = event_target:gk_raidee }
			}
		}

		gk_destroy_pop_result_tracker = yes
	}
}

# Bucket completed — offer escalation to side leader
# This = gk_cap_raider (side leader)
# From = raidee country
country_event = {
	id = gk_raiding_pops.500
	title = GK_RAIDING_ESCALATION_TITLE
	desc = {
		trigger = {
			event_target:gk_raid_tracker = {
				check_variable = { which = gk_escalation_tier value >= 4 }
			}
		}
		text = GK_RAIDING_ESCALATION_DESC_FINAL
	}
	desc = {
		trigger = {
			NOT = { has_country_flag = gk_raid_is_significant }
			has_country_flag = gk_escalation_has_intel
			event_target:gk_raid_tracker = {
				NOT = { check_variable = { which = gk_escalation_tier value >= 4 } }
			}
		}
		text = GK_RAIDING_ESCALATION_DESC_TRIVIAL_INTEL
	}
	desc = {
		trigger = {
			NOT = { has_country_flag = gk_raid_is_significant }
			NOT = { has_country_flag = gk_escalation_has_intel }
			event_target:gk_raid_tracker = {
				NOT = { check_variable = { which = gk_escalation_tier value >= 4 } }
			}
		}
		text = GK_RAIDING_ESCALATION_DESC_TRIVIAL
	}
	desc = {
		trigger = {
			has_country_flag = gk_escalation_has_intel
			event_target:gk_raid_tracker = {
				NOT = { check_variable = { which = gk_escalation_tier value >= 4 } }
			}
		}
		text = GK_RAIDING_ESCALATION_DESC_INTEL
	}
	desc = {
		trigger = {
			event_target:gk_raid_tracker = {
				NOT = { check_variable = { which = gk_escalation_tier value >= 4 } }
			}
		}
		text = GK_RAIDING_ESCALATION_DESC
	}
	picture = GFX_evt_ground_combat
	show_sound = event_alien_nature
	is_triggered_only = yes

	trigger = {
		# Only show if war still ongoing and cap still reached
		is_at_war_with = from
		from = {
			has_country_flag = gk_raid_cap_reached_by_@root
			NOT = { has_country_flag = gk_pop_raiding_terminated_by_@root }
		}
	}

	immediate = {
		save_event_target_as = gk_cap_raider
		from = {
			save_event_target_as = gk_raidee
		}
		gk_get_or_create_raid_tracker = yes
		# Compute next bucket yield for display: raidee pops * next tier rate
		set_variable = { which = gk_display_bucket_remaining value = from.trigger:count_owned_pop_amount }
		# Next tier rate lookup (current tier just completed, so next = tier + 1)
		set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_1 }
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 1 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_2 }
		}
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 2 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_3 }
		}
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 3 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_4 }
		}
		if = {
			limit = { event_target:gk_raid_tracker = { check_variable = { which = gk_escalation_tier value >= 4 } } }
			set_variable = { which = gk_display_next_rate value = @gk_bucket_rate_5 }
		}
		multiply_variable = {
			which = gk_display_bucket_remaining
			value = gk_display_next_rate
		}
		clear_variable = gk_display_next_rate
		# Compute dynamic influence cost adjustment: 25 * tier * (yield / raider_pops) / baseline
		# Round to nearest 25
		export_trigger_value_to_variable = {
			trigger = count_owned_pop_amount

# Tribute deposit waiting for raider to collect
# This = raider
# From = raidee
# Deposit amount stored as gk_tribute_deposit on raid tracker
# Type flags on tracker: gk_tribute_type_energy, gk_tribute_type_food,
#   gk_tribute_type_alloys, gk_tribute_type_split_energy, gk_tribute_type_split_food
country_event = {
	id = gk_raiding_pops.511
	title = GK_RAIDING_TRIBUTE_DEPOSIT_TITLE
	desc = GK_RAIDING_TRIBUTE_DEPOSIT_DESC
	picture = GFX_evt_treasure_chamber
	show_sound = event_celebration
	is_triggered_only = yes

	immediate = {
		save_event_target_as = gk_cap_raider
		from = { save_event_target_as = gk_raidee }
		gk_get_or_create_raid_tracker = yes
	}

	option = {
		name = GK_RAIDING_TRIBUTE_COLLECT
		# Deposit based on type
		if = {
			limit = { event_target:gk_raid_tracker = { has_leader_flag = gk_tribute_type_energy } }
			add_resource = {
				energy = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit
			}
		}
		else_if = {
			limit = { event_target:gk_raid_tracker = { has_leader_flag = gk_tribute_type_food } }
			add_resource = {
				food = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit
			}
		}
		else_if = {
			limit = { event_target:gk_raid_tracker = { has_leader_flag = gk_tribute_type_alloys } }
			add_resource = {
				alloys = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit
			}
		}
		else_if = {
			limit = { event_target:gk_raid_tracker = { has_leader_flag = gk_tribute_type_split_energy } }
			add_resource = {
				energy = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit
			}
			add_resource = {
				alloys = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit_alloy
			}
		}
		else_if = {
			limit = { event_target:gk_raid_tracker = { has_leader_flag = gk_tribute_type_split_food } }
			add_resource = {
				food = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit
			}
			add_resource = {
				alloys = 1
				mult = event_target:gk_raid_tracker.gk_tribute_deposit_alloy
			}
		}
		# Refund influence
		add_resource = {
			influence = 1
			mult = value:gk_escalation_influence_cost
		}
		hidden_effect = {
			event_target:gk_raid_tracker = {
				clear_variable = gk_tribute_deposit
				clear_variable = gk_tribute_deposit_alloy
				clear_variable = gk_tribute_base
				remove_leader_flag = gk_tribute_type_energy
				remove_leader_flag = gk_tribute_type_food
				remove_leader_flag = gk_tribute_type_alloys
				remove_leader_flag = gk_tribute_type_split_energy
				remove_leader_flag = gk_tribute_type_split_food
			}
		}
	}
}
