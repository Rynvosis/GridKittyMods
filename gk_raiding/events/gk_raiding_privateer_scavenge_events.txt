namespace = gk_raiding_privateers

# ──────────────────────────────────────────────────────────────────
# Debris Scavenging (700–709)
# ──────────────────────────────────────────────────────────────────
# Raider battles generate debris. The enclave auto-salvages it via
# pass_debris_ownership + salvage_ships_from_debris (vanilla paragon effect).
# Monthly poll on raider catches debris during raids; cleanup sweep catches the rest.

# 700 — Monthly: sweep raider debris during active raids
# This = raider country (from on_monthly_pulse_country)
country_event = {
	id = gk_raiding_privateers.700
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = gk_privateer_raiders
		count_available_debris = { count > 0 }
	}

	immediate = {
		# Find parent enclave
		random_country = {
			limit = {
				is_country_type = gk_privateer_enclave
				prev = { has_country_flag = gk_privateer_raider_of_@prev }
			}
			save_event_target_as = gk_privateer_enclave_country
		}

		if = {
			limit = { exists = event_target:gk_privateer_enclave_country }
			gk_sweep_raider_debris = yes
		}
	}
}

# 701 — Hidden: enclave processes a single debris field
# Fired from gk_sweep_raider_debris inside random_available_debris scope
# This = enclave country, From = debris
# from.controller = destroyed ship's controlling country
country_event = {
	id = gk_raiding_privateers.701
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		salvage_ships_from_debris = {
			CHANCE = 30
		}
	}
}
