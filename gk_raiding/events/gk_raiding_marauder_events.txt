# Enclave Founding (100–199)
# ──────────────────────────────────────────────────────────────────

# Execute creation — hidden
# This = Fleet
fleet_event = {
	id = gk_raiding_marauder.101
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		create_enclave_with_country = {
			PREFIX = gk_marauder
			EFFECT = gk_create_marauder_enclave_country
			TRUST = 30
		}
	}

	after = {
		event_target:gk_marauder_enclave_creator = {
			country_event = { id = gk_raiding_marauder.102 }
			set_country_flag = gk_is_marauder_patron
		}
	}
}

# Haven established — notification to patron
# This = Patron country

		SCOPE = event_target:gk_marauder_enclave_country
	}
	picture = GFX_evt_pirate_armada
	show_sound = event_ship_bridge
	is_triggered_only = yes

	option = {
		name = EXCELLENT
	}
}

# ──────────────────────────────────────────────────────────────────
# Diplomacy Window — Gatekeeper + Greeting (200–299)
# ──────────────────────────────────────────────────────────────────

# Gatekeeper — redirects on_custom_diplomacy to greeting
# This = player country, From = enclave country
country_event = {
	id = gk_raiding_marauder.200
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		from = {
			is_country_type = gk_marauder_enclave
		}
		NOT = {
			has_active_event = {
				gk_raiding_marauder.205
			}
		}
	}

	immediate = {
		from = {
			save_event_target_as = gk_marauder_enclave_country
		}
		country_event = { id = gk_raiding_marauder.205 }
	}
}
