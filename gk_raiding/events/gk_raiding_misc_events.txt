namespace = gk_raiding_misc

############################
# Machine Despoilers Setup #
############################
# This = country
# Converts 400 starting machine pops to organic battery pops
# Sets secondary species to grid amalgamation slavery
# Ensures a farming district on capital
country_event = {
	id = gk_raiding_misc.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_valid_civic = civic_gk_barbaric_despoilers_machine
	}

	immediate = {
		# 1. Find species by type (vanilla sometimes swaps founder identity)
		every_owned_species = {
			limit = { is_robotic = yes }
			save_event_target_as = gk_machine_species
		}
		every_owned_species = {
			limit = { is_robotic = no }
			save_event_target_as = gk_battery_species
		}

		# 2. Store total starting pop count
		capital_scope = {
			export_trigger_value_to_variable = {
				trigger = pop_amount
				variable = gk_starting_pops
			}
		}

		# 3. Set slavery on battery species
		event_target:gk_battery_species = {
			set_citizenship_type = {
				country = root
				type = citizenship_slavery
				cooldown = no
			}
			set_slavery_type = {
				country = root
				type = slavery_matrix
				cooldown = no
			}
		}

		# 4. Convert all pops to machines (clean slate, fixes vanilla founder swap)
		change_dominant_species = {
			species = event_target:gk_machine_species
			change_all = yes
		}

		# 5. Kill 400 machines and replace with battery pops
		capital_scope = {
			random_owned_pop_group = {
				limit = { is_same_species = event_target:gk_machine_species }
				kill_pop_group = {
					pop_group = this
					amount = 400
				}
			}
			create_pop_group = {
				species = event_target:gk_battery_species
				size = 400
			}
		}

		# 5. Enforce machine species as dominant
		change_dominant_species = {
			species = event_target:gk_machine_species
			change_all = no
		}

		# 6. Ensure a farming district for organic upkeep
		capital_scope = {
			if = {
				limit = { NOT = { has_district = district_farming } }
				add_district = district_farming
			}
			if = {
				limit = { has_district = district_generator }
				remove_district = district_generator
			}
		}

		# 7. Cleanup
		clear_variable = gk_starting_pops
	}
}

# ADDITIVE TOP-UP of vanilla's ship-kill trait rewards (vanilla paragon_2.1005, on on_ship_destroyed_perp).
# This is intentionally NOT an overwrite of paragon_2.1005 — we add an extra reward ON TOP so the mod stays
# compatible across patches. Gated on the Acquisitive Officers tradition (tr_gk_raiding_despoliation_5) and
# kills vs default/fallen empires, it grants an extra gk_despoliation_shipkill_trait_mult (+50%, x1.5 with
# Hell's Heart) of each trait's base reward for Corsair (energy), Shipbreaker (alloys) and Plunderer (trade),
# plus their _2 tiers. Do NOT "deduplicate" this against vanilla — the extra reward is by design.
# (Plunderer note: vanilla's native Plunderer reward does not scale with Hell's Heart; our top-up does, for
# uniformity with the Corsair/Shipbreaker top-ups.)
# Fired by on_ship_destroyed_perp
# This = owner of winning ship
# From = owner of destroyed ship
# FromFrom = winning ship
# FromFromFrom = destroyed ship
country_event = {
	id = gk_raiding_misc.111
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_tradition = tr_gk_raiding_despoliation_5
		FROM = {
			is_default_or_fallen = yes
		}
	}

	immediate = {
		FROMFROM = {
			fleet.leader? = {
				if = {
					limit = {
						OR = {
							has_trait = leader_trait_corsair
							has_trait = leader_trait_corsair_2
						}
					}
					if = {
						limit = {
							has_trait = leader_trait_corsair
						}
						ROOT = {
							add_resource = {
								energy = @trait_corsair_amt
								mult = value:gk_despoliation_shipkill_trait_mult
							}
						}
					}
					else = {
						ROOT = {
							add_resource = {
								energy = @trait_corsair_2_amt
								mult = value:gk_despoliation_shipkill_trait_mult
							}
						}
					}
				}

				if = {
					limit = {
						OR = {
							has_trait = leader_trait_shipbreaker
							has_trait = leader_trait_shipbreaker_2
						}
					}
					if = {
						limit = {
							has_trait = leader_trait_shipbreaker
						}
						ROOT = {
							add_resource = {
								alloys = @trait_shipbreaker_amt
								mult = value:gk_despoliation_shipkill_trait_mult
							}
						}
					}
					else = {
						ROOT = {
							add_resource = {
								alloys = @trait_shipbreaker_2_amt
								mult = value:gk_despoliation_shipkill_trait_mult
							}
						}
					}
				}

				# Plunderer top-up (trade). Vanilla's native Plunderer reward isn't Hell's-Heart-scaled; ours is.
				if = {
					limit = {
						OR = {
							has_trait = leader_trait_plunderer
							has_trait = leader_trait_plunderer_2
						}
					}
					if = {
						limit = {
							has_trait = leader_trait_plunderer
						}
						ROOT = {
							add_resource = {
								trade = @trait_plunderer_amt
								mult = value:gk_despoliation_shipkill_trait_mult
							}
						}
					}
					else = {
						ROOT = {
							add_resource = {
								trade = @trait_plunderer_2_amt
								mult = value:gk_despoliation_shipkill_trait_mult
							}
						}
					}
				}
			}
		}
	}
}
