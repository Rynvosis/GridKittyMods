# gk_raiding_spawn_marauder            — planet scope, creates dormant marauder + builds out system as capital
# gk_raiding_extend_marauder            — planet scope, claims/builds current system as aux. Optional RANGE for radius.
#   effect gk_raiding_extend_marauder = yes
#   effect gk_raiding_extend_marauder = { RANGE = 4 }
# gk_raiding_awaken_marauder_as_khan    — country scope, awakens this country as the Great Khan.
#                                         Inlines the marauder.500 immediate body, skipping the
#                                         marauder_1/2/3 slot branches (which look up rally points
#                                         via vanilla star/planet flags we don't carry).
#   effect event_target:mk_main = { gk_raiding_awaken_marauder_as_khan = yes }
# gk_raiding_seed_horde_planets         — for each system owned by event_target:gk_utils_marauder,
#                                         1/2 chance to convert a basic planet into a tomb (50%) /
#                                         gaia (25%) / relic (25%) world. After each successful seed,
#                                         50% chain chance to seed another planet in the same system.
#                                         Chain unrolled up to 5 total passes (~33% colony rate, ~33%
#                                         tomb rate, mean ~0.67 trophies per system).
#   effect gk_raiding_seed_horde_planets = yes
# gk_raiding_colonize_horde_planets     — for each habitable planet owned by event_target:gk_utils_marauder,
#                                         create a colony with vanilla advanced-AI infrastructure
#                                         (capital building + a few districts via game_start.21/.22),
#                                         then 1500 main-species pops + 1000 pops from a random
#                                         other-marauder country's species.
#   effect gk_raiding_colonize_horde_planets = yes
# gk_raiding_spawn_marauder_attack_fleet — spawns one mobile attack fleet (11 Outrider + 5 Lancer +
#                                         4 Void_Champion + 1 Ancestral_Glory, with admiral, no
#                                         garrison flag) at event_target:gk_utils_marauder's capital
#                                         by re-pointing raider_khanate / marauder_rally_point at the
#                                         active marauder, calling vanilla create_marauder_fleet, then
#                                         restoring the originals.
#   effect gk_raiding_spawn_marauder_attack_fleet = yes
#
# Lesser-khan satrapy bind: when binding the lesser khan as a satrapy of the Khan, use the new
# specialist satrapy preset rather than vanilla preset_satrapy. The new preset uses
# specialist_satrapy (a 4th specialist subject type peer to bulwark/scholarium/prospectorium)
# defined in gk_raiding/common/specialist_subject_types/. Console paste:
#   effect event_target:mk_lesser = { ... set_subject_of = { who = event_target:raider_khanate preset = preset_satrapy_warrior } ... }
#
# Note on naming: these effects were originally `gk_utils_*` in gk_utils. The event-target and
# country-flag names below (gk_utils_marauder, gk_utils_marauder_rally, gk_utils_local_focus,
# gk_utils_main_species, gk_utils_secondary_species, gk_utils_saved_khanate, gk_utils_saved_rally,
# gk_utils_spawned_marauder) are kept as `gk_utils_*` for save-compat — they're stored values in
# active campaigns. Don't rename them without a save-migration plan.

gk_raiding_awaken_marauder_as_khan = {
	set_country_type = awakened_marauders
	set_global_flag = marauder_crisis_started
	set_global_flag = marauder_crisis_ongoing
	save_global_event_target_as = raider_khanate
	capital_scope = {
		save_global_event_target_as = marauder_rally_point
		set_planet_flag = marauder_rally_point
	}
	change_government = {
		civics = { civic = civic_great_khans_vision }
	}
	set_name = random
	change_country_flag = {
		icon = { category = "pirate" file = "flag_pirate_11.dds" }
		background = { category = "backgrounds" file = "00_solid.dds" }
		colors = { "red" "red" "null" "null" }
	}
	create_leader = {
		class = commander
		species = owner_main_species
		name = random
		skill = 10
		leader_age_min = 22
		leader_age_max = 40
		randomize_traits = no
		traits = {
			trait = leader_trait_master_gunner
			trait = subclass_commander_admiral
			trait = leader_trait_gunship_specialization_2
			trait = leader_trait_wrecker_2
			trait = leader_trait_reckless_2
			trait = leader_trait_artillerist_2
			trait = leader_trait_demolisher_2
		}
	}
	last_created_leader = {
		set_leader_flag = great_khan
		save_global_event_target_as = great_khan_ruler
	}
	# Council trait must be added AFTER ruler assignment — added during create_leader
	# it errors with councilor_trait_not_allowed because the leader isn't on the council yet.
	assign_leader = last_created_leader
	event_target:great_khan_ruler = {
		add_trait = { trait = leader_trait_ruler_great_khan }
	}
	set_timed_global_flag = { flag = khan_no_die days = 7200 }
	country_event = { id = marauder.514 days = 10800 random = 5000 }
	clone_leader = {
		target = event_target:great_khan_ruler
		class = commander
		tier = leader_tier_legendary
		skill = 10
		randomize_traits = no
		effect = {
			remove_trait = leader_trait_ruler_great_khan
			add_trait = { trait = leader_trait_great_khan }
			set_leader_flag = great_khan
			save_global_event_target_as = great_khan
		}
	}
	create_1st_khan_fleet = yes
	create_marauder_armies = yes
	create_marauder_armies = yes
	create_marauder_armies = yes
	create_marauder_armies = yes
	create_marauder_constructors = yes
	create_marauder_constructors = yes
	create_marauder_constructors = yes
	country_event = { id = marauder.520 days = 180 }
	observer_event = { id = observer.13 }
	every_country = {
		limit = { is_country_type = default }
		establish_communications_no_message = event_target:raider_khanate
		country_event = { id = marauder.501 }
	}
}


gk_raiding_extend_marauder = {
	solar_system = {
		if = {
			limit = { NOT = { has_star_flag = marauder_system } }
			gk_raiding_buildout_marauder_aux_system = yes
		}
		[[RANGE]
			every_system = {
				limit = {
					distance = {
						source = prev
						max_jumps = $RANGE$
						use_bypasses = no
					}
					NOT = { has_star_flag = marauder_system }
				}
				gk_raiding_buildout_marauder_aux_system = yes
			}
		]
	}
}


# Mirrors marauder_X_1 (capital) from common/solar_system_initializers/marauder_initializers.txt:
# starbase, 8 Warrior_Freehold patrols, central Ancestral_Glory garrison, BIG horde fleet,
# per-planet Void_Dwelling defenses.
gk_raiding_buildout_marauder_capital_system = {
	set_star_flag = marauder_system

	if = {
		limit = { exists = starbase }
		starbase = { set_owner = event_target:gk_utils_marauder }
	}
	if = {
		limit = { NOT = { exists = starbase } }
		create_starbase = {
			owner = event_target:gk_utils_marauder
			size = starbase_marauder
		}
	}

	random_system_planet = {
		limit = { is_star = yes }
		save_event_target_as = gk_utils_local_focus
	}

	while = {
		count = 8
		create_fleet = {
			effect = {
				set_owner = event_target:gk_utils_marauder
				create_ship = {
					name = random
					design = "NAME_Warrior_Freehold"
					graphical_culture = "pirate_01"
				}
				set_location = {
					target = event_target:gk_utils_local_focus
					distance = 230
					angle = random
				}
			}
		}
	}

	create_fleet = {
		effect = {
			set_owner = event_target:gk_utils_marauder
			create_ship = {
				name = random
				design = "NAME_Ancestral_Glory"
				graphical_culture = "pirate_01"
			}

			set_fleet_stance = aggressive

			set_aggro_range = 250
			set_location = {

				distance = 20
				angle = random
			}
		}

	}

		effect = {
			set_owner = event_target:gk_utils_marauder

				create_ship = {
					name = random
					design = "NAME_Outrider"
					graphical_culture = "pirate_01"

			}
			while = {
				count = 14
				create_ship = {
					name = random
					design = "NAME_Lancer"
					graphical_culture = "pirate_01"
				}
			}
			while = {
				count = 8
				create_ship = {
					name = random
					design = "NAME_Void_Champion"
					graphical_culture = "pirate_01"
				}
			}
			set_formation_scale = 2
			set_fleet_stance = aggressive
			set_aggro_range_measure_from = self
			set_aggro_range = 250
			set_location = {
				target = event_target:gk_utils_local_focus
				distance = 20
				angle = random
			}
		}
		settings = { garrison = yes }

gk_raiding_seed_horde_planets = {
	every_system = {
		limit = {
			starbase = { is_owned_by = event_target:gk_utils_marauder }
			NOR = {
				is_star_class = sc_neutron_star
				is_star_class = sc_pulsar
				is_star_class = sc_black_hole
			}
		}
		# Defensive: clear stale chain flag from a prior interrupted run.
		remove_star_flag = gk_seed_chain
		# Pass 1 always runs; subsequent passes gate on the chain flag set by the previous pass.
		gk_raiding_seed_horde_planet_pass = yes
		if = {
			limit = { has_star_flag = gk_seed_chain }
			remove_star_flag = gk_seed_chain
			gk_raiding_seed_horde_planet_pass = yes
		}
		if = {
			limit = { has_star_flag = gk_seed_chain }
			remove_star_flag = gk_seed_chain
			gk_raiding_seed_horde_planet_pass = yes
		}
		if = {
			limit = { has_star_flag = gk_seed_chain }
			remove_star_flag = gk_seed_chain
			gk_raiding_seed_horde_planet_pass = yes
		}
		if = {
			limit = { has_star_flag = gk_seed_chain }
			remove_star_flag = gk_seed_chain
			gk_raiding_seed_horde_planet_pass = yes
		}
		# Final cleanup in case pass 5 set the flag but we ran out of unrolled iterations.
		remove_star_flag = gk_seed_chain
	}
}

# Single seeding pass. Entry: system scope.
# 50% chance to convert a non-trophy planet to tomb / gaia / relic (weight 2/1/1).
# On successful seed, 50% chance to set gk_seed_chain on the system, allowing another pass.
gk_raiding_seed_horde_planet_pass = {
	random = {
		chance = 50
		random_system_planet = {
			limit = {
				NOT = { is_star = yes }
				is_moon = no
				NOR = {
					is_planet_class = pc_asteroid
					is_planet_class = pc_gas_giant
					is_planet_class = pc_nuked
					is_planet_class = pc_gaia
					is_planet_class = pc_relic
				}
			}
			random_list = {
				2 = {
					change_pc = pc_nuked
					random_list = {
						1 = { set_planet_size = 16 }
						1 = { set_planet_size = 18 }
						1 = { set_planet_size = 20 }
						1 = { set_planet_size = 22 }
						1 = { set_planet_size = 25 }
					}
				}
				1 = {
					change_pc = pc_gaia
					random_list = {
						1 = { set_planet_size = 12 }
						1 = { set_planet_size = 15 }
						1 = { set_planet_size = 17 }
						1 = { set_planet_size = 20 }
						1 = { set_planet_size = 22 }
					}
				}
				1 = {
					change_pc = pc_relic
					random_list = {
						1 = { set_planet_size = 8 }
						1 = { set_planet_size = 10 }
						1 = { set_planet_size = 12 }
						1 = { set_planet_size = 14 }
						1 = { set_planet_size = 15 }
					}
				}
			}
			# We're in planet scope; chain flag lives on the system.
			solar_system = {
				random = {
					chance = 50
					set_star_flag = gk_seed_chain
				}
			}
		}
	}
}


gk_raiding_colonize_horde_planets = {
	event_target:gk_utils_marauder = {
		owner_main_species = { save_global_event_target_as = gk_utils_main_species }
	}
	every_system = {
		limit = { starbase = { is_owned_by = event_target:gk_utils_marauder } }
		every_system_planet = {
			limit = {
				NOT = { exists = owner }
				OR = {
					is_planet_class = pc_continental
					is_planet_class = pc_arid
					is_planet_class = pc_desert
					is_planet_class = pc_alpine
					is_planet_class = pc_arctic
					is_planet_class = pc_tropical
					is_planet_class = pc_savannah
					is_planet_class = pc_ocean
					is_planet_class = pc_tundra
					is_planet_class = pc_gaia
					is_planet_class = pc_relic
				}
			}
			# Class-aware deposits/modifiers. Done BEFORE create_colony because
			# relic_world_deposits does reset_planet which would wipe a freshly-built colony.
			if = {
				limit = { is_planet_class = pc_relic }
				relic_world_deposits = yes
			}
			else = {
				reroll_deposits = yes
				reroll_planet_modifiers = yes
			}
			create_colony = {
				owner = event_target:gk_utils_marauder
				species = event_target:gk_utils_main_species
			}
			# Vanilla advanced-colony infrastructure: .21 swaps colony shelter → capital building,
			# .22 lays down 4-5 districts. Skip .23 (vanilla pops) — we want the user-specified counts below.
			planet_event = { id = game_start.21 days = 0 }
			planet_event = { id = game_start.22 days = 0 }
			create_pop_group = {
				species = event_target:gk_utils_main_species
				size = 1500
			}
			random_country = {
				limit = {
					OR = {
						is_country_type = dormant_marauders
						is_country_type = awakened_marauders
					}
					NOT = { is_same_value = event_target:gk_utils_marauder }
				}
				owner_main_species = { save_global_event_target_as = gk_utils_secondary_species }
			}
			create_pop_group = {
				species = event_target:gk_utils_secondary_species
				size = 1000
			}
		}
	}
}


gk_raiding_spawn_marauder_attack_fleet = {
	# Stash current vanilla event targets so we can restore after the call.
	if = {
		limit = { exists = event_target:raider_khanate }
		event_target:raider_khanate = { save_global_event_target_as = gk_utils_saved_khanate }
	}
	if = {
		limit = { exists = event_target:marauder_rally_point }
		event_target:marauder_rally_point = { save_global_event_target_as = gk_utils_saved_rally }
	}
	# Re-point at the active marauder + its capital planet.
	event_target:gk_utils_marauder = {
		save_global_event_target_as = raider_khanate
		capital_scope = {
			save_global_event_target_as = marauder_rally_point
		}
	}
	create_marauder_fleet = yes
	# Restore originals so Khan-related vanilla code keeps seeing the right targets.
	if = {
		limit = { exists = event_target:gk_utils_saved_khanate }
		event_target:gk_utils_saved_khanate = { save_global_event_target_as = raider_khanate }
	}
	if = {
		limit = { exists = event_target:gk_utils_saved_rally }
		event_target:gk_utils_saved_rally = { save_global_event_target_as = marauder_rally_point }
	}
}
