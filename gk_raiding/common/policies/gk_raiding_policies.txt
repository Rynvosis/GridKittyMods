# GK Raiding Policies
# Determines raiding focus and grants associated modifiers
# gk_pillage_efficiency = multiplier on resources gained from pillaging
# Resource pillaging is driven by ongoing battlefield damage (see gk_pillage_effects.txt)
# gk_pillage_efficiency = multiplier on resources gained from pillaging

gk_raiding_focus = {
	allow = {
		is_at_war = no # Not allowed to change policies in this group while at war
	}

	potential = {
		OR = {
			has_valid_civic = civic_barbaric_despoilers
			has_valid_civic = civic_crusader_spirit_corporate
			has_ascension_perk = ap_nihilistic_acquisition
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_emancipators_megacorp
			has_valid_civic = civic_gk_robo_liberators
			has_valid_civic = civic_gk_robo_liberators_megacorp
			has_valid_civic = civic_gk_robo_liberators_machine
			has_valid_civic = civic_gk_barbaric_despoilers_machine
			has_valid_civic = civic_gk_barbaric_despoilers_megacorp
		}
	}

	option = {
		name = "gk_raiding_focus_plunder"

		valid = {
			NOR = {
				has_valid_civic = civic_gk_emancipators
				has_valid_civic = civic_gk_emancipators_megacorp
				has_valid_civic = civic_gk_robo_liberators
				has_valid_civic = civic_gk_robo_liberators_megacorp
				has_valid_civic = civic_gk_robo_liberators_machine
			}
		}

		policy_flags = {
			gk_focus_plunder
		}

		AI_weight = {
			factor = 2
		}
	}

	option = {
		name = "gk_raiding_focus_no_raiding"

		valid = {
			NOR = {
				has_valid_civic = civic_gk_emancipators
				has_valid_civic = civic_gk_emancipators_megacorp
				has_valid_civic = civic_gk_robo_liberators
				has_valid_civic = civic_gk_robo_liberators_megacorp
				has_valid_civic = civic_gk_robo_liberators_machine
			}
		}

		policy_flags = {
			gk_focus_no_pop_raiding
		}

		AI_weight = {
			factor = 1
			modifier = {
				factor = 2
				has_ethic = ethic_fanatic_militarist
			}
		}
	}


	option = {
		name = "gk_raiding_focus_emancipation"

		valid = {
			NOR = {
				has_valid_civic = civic_gk_robo_liberators
				has_valid_civic = civic_gk_robo_liberators_megacorp
				has_valid_civic = civic_gk_robo_liberators_machine
			}
			OR = {
				has_valid_civic = civic_gk_emancipators
				has_valid_civic = civic_gk_emancipators_megacorp
				has_ethic = ethic_egalitarian
				has_ethic = ethic_fanatic_egalitarian
			}
		}

		policy_flags = {
			gk_focus_emancipation
		}

		AI_weight = {
			factor = 2
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_egalitarian
			}
			modifier = {
				factor = 2
				has_valid_civic = civic_gk_emancipators
			}
		}
	}
	option = {
		name = "gk_raiding_focus_robo_liberation"

		valid = {
			OR = {
				has_valid_civic = civic_gk_robo_liberators
				has_valid_civic = civic_gk_robo_liberators_megacorp
				has_valid_civic = civic_gk_robo_liberators_machine
			}
		}

		policy_flags = {
			gk_focus_robo_liberation
		}

		AI_weight = {
			factor = 10
			modifier = {
				factor = 2
				has_valid_civic = civic_gk_robo_liberators
			}
		}
	}
}

# Policy to determine if robots can be raided

	allow = {
		is_at_war = no # Not allowed to change policies in this group while at war

				factor = 3

		policy_flags = {

				has_valid_civic = civic_gk_robo_liberators_megacorp
				has_valid_civic = civic_gk_robo_liberators_machine
			}
		}

		AI_weight = {
			factor = 1
			modifier = {
				factor = 5
				has_ethic = ethic_fanatic_materialist
			}
		}
