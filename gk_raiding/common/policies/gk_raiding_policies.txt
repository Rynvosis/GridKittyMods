# GK Raiding Policies
# Determines raiding focus and grants associated modifiers
# gk_pop_raiding_limit = percentage of enemy pops you can take
# gk_pillage_efficiency = multiplier on resources gained from pillaging
# gk_max_pillage_tiers = additional times you can pillage a planet

gk_raiding_focus = {
	allow = {
		is_at_war = no # Not allowed to change policies in this group while at war
	}

	potential = {
		OR = {
			has_valid_civic = civic_barbaric_despoilers
			has_valid_civic = civic_crusader_spirit_corporate
			has_ascension_perk = ap_nihilistic_acquisition
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_robo_liberators
		}
	}
	
	option = {
		name = "gk_raiding_focus_plunder"

		valid = {
			NOR = {
				has_valid_civic = civic_gk_emancipators
				has_valid_civic = civic_gk_robo_liberators
			}
		}
		
		policy_flags = {
			gk_focus_plunder
		}
		
		# 10% pop limit, 2 pillage tiers
		modifier = {
			gk_pop_raiding_limit = 1.0
			gk_max_pillage_tiers = 2
		}
		
		AI_weight = {
			factor = 2
		}
	}
	
	option = {
		name = "gk_raiding_focus_despoliation"

		valid = {
			NOR = {
				has_valid_civic = civic_gk_emancipators
				has_valid_civic = civic_gk_robo_liberators
			}
		}
		
		policy_flags = {
			gk_focus_despoliation
		}
		
		# No pop raiding, 6 pillage tiers, +50% pillage efficiency
		modifier = {
			gk_pillage_efficiency = 1.0
			gk_max_pillage_tiers = 6
		}
		
		AI_weight = {
			factor = 1
			modifier = {
				factor = 2
				has_ethic = ethic_fanatic_militarist
			}
		}
	}
	
	
	option = {
		name = "gk_raiding_focus_emancipation"
		
		valid = {
			NOT = { has_valid_civic = civic_gk_robo_liberators }
			OR = {
				has_valid_civic = civic_gk_emancipators
				has_ethic = ethic_fanatic_egalitarian
			}
		}
		
		policy_flags = {
			gk_focus_emancipation
		}
		
		modifier = {
			gk_pop_raiding_limit = 1.0
		}
		
		AI_weight = {
			factor = 2
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_egalitarian
			}
			modifier = {
				factor = 2
				has_valid_civic = civic_gk_emancipators
			}
		}
	}
	option = {
		name = "gk_raiding_focus_robo_liberation"
		
		valid = {
			has_valid_civic = civic_gk_robo_liberators
		}
		
		policy_flags = {
			gk_focus_robo_liberation
		}
		
		# 15% limit, robots only
		modifier = {
			gk_pop_raiding_limit = 2.0
		}
		
		AI_weight = {
			factor = 10
			modifier = {
				factor = 2
				has_valid_civic = civic_gk_robo_liberators
			}
		}
	}
}

# Policy to determine if robots can be raided
# Hidden for robo-liberators (they are forced to robots only)
gk_robot_raiding = {
	allow = {
		is_at_war = no # Not allowed to change policies in this group while at war
	}

	potential = {
		OR = {
			has_valid_civic = civic_barbaric_despoilers
			has_valid_civic = civic_crusader_spirit_corporate
			has_ascension_perk = ap_nihilistic_acquisition
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_robo_liberators
		}
	}
	
	option = {
		name = "gk_robot_raiding_allowed"
		
		policy_flags = {
			gk_allow_robot_raiding
		}

		valid = {
			NOT = { has_valid_civic = civic_gk_robo_liberators }
		}
		
		AI_weight = {
			factor = 2
			modifier = {
				factor = 3
				has_ethic = ethic_materialist
			}
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_materialist
			}
		}
	}
	
	option = {
		name = "gk_robot_raiding_forbidden"
		
		policy_flags = {
			gk_forbid_robot_raiding
		}

		valid = {
			NOT = { has_valid_civic = civic_gk_robo_liberators }
		}
		
		AI_weight = {
			factor = 1
			modifier = {
				factor = 3
				has_ethic = ethic_spiritualist
			}
			modifier = {
				factor = 3
				has_ethic = ethic_fanatic_spiritualist
			}
		}
	}
	
	option = {
		name = "gk_robot_raiding_only"
		
		policy_flags = {
			gk_robots_only_raiding
		}

		potential = {
			has_valid_civic = civic_gk_robo_liberators
		}
		
		AI_weight = {
			factor = 1
			modifier = {
				factor = 5
				has_ethic = ethic_fanatic_materialist
			}
		}
	}
}
