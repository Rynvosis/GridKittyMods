# RAIDING WAR       #
#####################
# General raiding: pops + resources, effectiveness based on modifiers
# Bombardment/invasion cap: 20%, Surrender cap: 10%
wg_plunder = {
	casus_belli = cb_despoliation
	hide_if_no_cb = yes
	total_war = no

	galactic_empire_joins_defender = {
		from = { is_galactic_emperor = yes }
		imperial_authority >= 100
	}

	surrender_acceptance = -75

	potential = {
		has_total_war_cb = no
		gk_can_use_raiding_cb = yes
		NOT = { has_policy_flag = gk_focus_no_pop_raiding }
		NOT = { is_in_federation_with = from }
		NAND = {
			is_galactic_emperor = yes
			from = { is_galactic_community_member = yes }
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_plunder_accept_tooltip
		hidden_effect = {
			# Set up event targets
			save_event_target_as = gk_raider
			from = { save_event_target_as = gk_raidee }
			# Set surrender cap rate (10% base, scaled by raiding capacity modifier)
			set_variable = { which = gk_cap_rate value = 0.10 }
			set_variable = { which = gk_rate_mult_temp value = value:gk_raiding_capacity_mult }
			multiply_variable = { which = gk_cap_rate value = gk_rate_mult_temp }
			clear_variable = gk_rate_mult_temp
			# Set pillage context for raiding war (2 tiers, normal yield)
			set_variable = { which = gk_effective_max_tiers value = 2 }
			set_variable = { which = gk_pillage_resource_mult value = 1 }
			# Pillage and abduct (all planets, all sapient pops per robot policy)
			gk_pillage_on_surrender = yes
			gk_abduct_on_surrender = yes
			# Fire reward collection event with raidee as from
			country_event = {
				id = gk_raiding.500
				scopes = { from = event_target:gk_raidee }
			}
		}
	}

	ai_weight = {
		weight = 100
	}
}

# PILLAGE WAR       #
#####################
# Resource-only pillaging with bonus tiers, no pop capture
wg_plunder_raid = {
	casus_belli = cb_pirate_raid
	hide_if_no_cb = yes
	total_war = no

	galactic_empire_joins_defender = {
		from = { is_galactic_emperor = yes }
		imperial_authority >= 100
	}

	surrender_acceptance = -50

	potential = {
		has_total_war_cb = no
		OR = {
			has_valid_civic = civic_crusader_spirit_corporate
			has_country_flag = unlocked_pirate_raid_cb
			has_ascension_perk = ap_nihilistic_acquisition
			has_valid_civic = civic_barbaric_despoilers
			has_valid_civic = civic_gk_barbaric_despoilers_machine
			has_valid_civic = civic_gk_barbaric_despoilers_megacorp
		}
		NOT = { is_in_federation_with = from }
		NAND = {
			is_galactic_emperor = yes
			from = { is_galactic_community_member = yes }
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_plunder_raid_accept_tooltip
		hidden_effect = {
			# Set up event targets
			save_event_target_as = gk_raider
			from = { save_event_target_as = gk_raidee }
			# Set pillage context for pillage war (6 tiers, doubled yield)
			set_variable = { which = gk_effective_max_tiers value = 6 }
			set_variable = { which = gk_pillage_resource_mult value = 2 }
			# Pillage only - no pop capture
			gk_pillage_on_surrender = yes
			# Fire pillage surrender event
			country_event = {
				id = gk_raiding.900
				scopes = { from = event_target:gk_raidee }
			}
		}
	}

	ai_weight = {
		weight = 50
		modifier = {
			factor = 2
			has_valid_civic = civic_crusader_spirit_corporate
		}
	}
}
