# GK Raiding Pop War Goals

@gk_reaping_cap_rate = 100

#####################
# REAPING WAR       #
#####################
# Unlimited pop capture, no resource pillaging
# Bombardment/invasion cap: unlimited, Surrender cap: all
wg_gk_reaping = {
	casus_belli = cb_despoliation
	hide_if_no_cb = no
	total_war = no

	galactic_empire_joins_defender = {
		from = { is_galactic_emperor = yes }
		imperial_authority >= 100
	}

	surrender_acceptance = -1000

	forbidden_peace_offers = {
		surrender = ""
	}

	potential = {
		has_total_war_cb = no
		gk_can_use_reaping_cb = yes
		NOT = { is_in_federation_with = from }
		NAND = {
			is_galactic_emperor = yes
			from = { is_galactic_community_member = yes }
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_gk_reaping_accept_tooltip
		hidden_effect = {
			# Set up event targets
			save_event_target_as = gk_raider
			from = { save_event_target_as = gk_raidee }
			# Abduct only - no pillaging (unlimited cap for reaping)
			gk_abduct_on_surrender = { RATE = @gk_reaping_cap_rate }
			# Fire reaping surrender event
			country_event = {
				id = gk_raiding.700
				scopes = { from = event_target:gk_raidee }
			}
		}
	}

	ai_weight = {
		weight = 75
		modifier = {
			factor = 2
			OR = {
				has_valid_civic = civic_barbaric_despoilers
				has_valid_civic = civic_gk_barbaric_despoilers_machine
				has_valid_civic = civic_gk_barbaric_despoilers_megacorp
			}
		}
	}
}

	}

			fail_text = gk_requires_enemy_has_slaves
			from = {
				is_country_type = default
				any_owned_planet = {
					any_owned_pop_group = {
						is_enslaved = yes
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {

				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_gk_emancipation_accept_tooltip
		hidden_effect = {
			# Set up event targets
			save_event_target_as = gk_raider
			from = { save_event_target_as = gk_raidee }
			# Abduct only (enslaved pops via war goal check)
			# Emancipation civic provides +50% gk_raiding_capacity, making effective rate ~15%
			gk_abduct_on_surrender = { RATE = value:gk_surrender_cap_rate }
			# Fire reward collection event with raidee as from
			country_event = {
				id = gk_raiding.400
				scopes = { from = event_target:gk_raidee }
			}
		}
	}

	on_status_quo = {
		custom_tooltip = wg_gk_emancipation_status_quo_tooltip
		# Keep what you've already taken
	}

	ai_weight = {
		weight = 50
		modifier = {
			factor = 3
			has_ethic = ethic_egalitarian
		}
		modifier = {
			factor = 3
			has_ethic = ethic_fanatic_egalitarian
		}
	}
}
