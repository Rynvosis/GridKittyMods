# Forward Raiding Bases: unlocks the starbase building (gating lives in the building potential).
tr_gk_raiding_despoliation_2 = {
	possible = {
		has_tradition = tr_gk_raiding_despoliation_1
	}

	custom_tooltip = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_2_EFFECT

	# Void Reavers use pirate waystations instead of the raiding base building.
	tradition_swap = {
		name = tr_gk_raiding_despoliation_2_void_reavers
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			has_valid_civic = civic_void_reavers
		}
		# The +50% piracy is applied by the nomads.5010 overwrite
		# (events/zz_gk_raiding_void_reaver_piracy_OW.txt), gated on this tradition — NOT a modifier
		# (starbase_collected_system_resources_mult would buff ALL of a starbase's resource collection,
		# not the pirate skim).
		custom_tooltip = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_2_VOID_REAVERS_EFFECT
		weight = {
			factor = 1
		}
	}

	# Other nomad empires (not Void Reavers) get pirate waystation MODULE access via this tradition (the
	# can_build_pirate_waystation_module override), instead of the starbase building. Evaluated AFTER the
	# Void Reavers swap, so a Void Reaver matches that one first.
	tradition_swap = {
		name = tr_gk_raiding_despoliation_2_nomad
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			is_nomadic = yes
		}
		custom_tooltip = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_2_NOMAD_EFFECT
		weight = {
			factor = 1
		}
	}

	ai_weight = {
		factor = 1000
	}
}

# Restless Raiders: stability swings based on recent war activity (flags set by events).
tr_gk_raiding_despoliation_3 = {
	possible = {
		has_tradition = tr_gk_raiding_despoliation_2
	}

	on_enabled = {
		hidden_effect = {
			set_timed_country_flag = { flag = gk_rr_war_window days = @gk_rr_window_days }
		}
	}

	triggered_modifier = {
		potential = {
			has_country_flag = gk_rr_recent_victory
		}
		planet_stability_add = 10
		pop_amenities_usage_mult = -0.15
	}

	triggered_modifier = {
		potential = {
			NOT = { has_country_flag = gk_rr_recent_victory }
			OR = {
				is_at_war = yes
				has_country_flag = gk_rr_recent_war
			}
		}
		planet_stability_add = 5
	}

	triggered_modifier = {
		# recent_war / recent_victory are always co-set with gk_rr_war_window (which lasts longer), so
		# NOT war_window already implies neither recent flag is set — these two checks suffice.
		potential = {
			is_at_war = no
			NOT = { has_country_flag = gk_rr_war_window }
		}
		planet_stability_add = -5
	}

	custom_tooltip = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_3_EFFECT

	ai_weight = {
		factor = 1000
	}
}
