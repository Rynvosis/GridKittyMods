tr_gk_raiding_despoliation_4 = {
	# Base: no scavenger, no Grand Archive
	custom_tooltip_with_modifiers = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_4_EFFECT

	# Swap: has Grand Archive, no scavenger
	tradition_swap = {
		name = tr_gk_raiding_despoliation_4_archive
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			has_grand_archive_dlc = yes
			NOT = {
				OR = {
					has_valid_civic = civic_scavengers
					has_valid_civic = civic_corporate_scavengers
				}
			}
		}
		on_enabled = {
			give_technology = { tech = tech_boarding_cables message = no }
		}
		custom_tooltip_with_modifiers = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_4_ARCHIVE_EFFECT

		weight = {
			factor = 1
		}
	}

	# Swap: has scavenger, no Grand Archive
	tradition_swap = {
		name = tr_gk_raiding_despoliation_4_scavenger
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			OR = {
				has_valid_civic = civic_scavengers
				has_valid_civic = civic_corporate_scavengers
			}
			NOT = { has_grand_archive_dlc = yes }
		}
		modifier = {
			salvage_chance_add = 0.10
		}
		custom_tooltip_with_modifiers = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_4_EFFECT

		weight = {
			factor = 1
		}
	}

	# Swap: has scavenger AND Grand Archive
	tradition_swap = {
		name = tr_gk_raiding_despoliation_4_scavenger_archive
		inherit_name = yes
		inherit_icon = yes
		trigger = {
			OR = {
				has_valid_civic = civic_scavengers
				has_valid_civic = civic_corporate_scavengers
			}
			has_grand_archive_dlc = yes
		}
		on_enabled = {
			give_technology = { tech = tech_boarding_cables message = no }
		}
		modifier = {
			salvage_chance_add = 0.10
		}
		custom_tooltip_with_modifiers = GK_RAIDING_TR_GK_RAIDING_DESPOLIATION_4_ARCHIVE_EFFECT

		weight = {
			factor = 1
		}
	}

	ai_weight = {
		factor = 1000
	}
}
