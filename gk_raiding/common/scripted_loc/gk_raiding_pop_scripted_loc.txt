# Dynamic press tooltip: normal vs final tier
defined_text = {
	name = GkPressTooltip
	text = {
		trigger = {
			exists = event_target:gk_raid_tracker
			event_target:gk_raid_tracker = {
				check_variable = { which = gk_escalation_tier value >= 4 }
			}
		}
		localization_key = GK_RAIDING_ESCALATION_PRESS_FINAL_TOOLTIP
	}
	text = {
		trigger = { always = yes }
		localization_key = GK_RAIDING_ESCALATION_PRESS_NORMAL_TOOLTIP
	}
}
