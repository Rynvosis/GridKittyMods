# GK Raiding Script Values

@gk_surrender_base_cap_rate = 0.10
@gk_invasion_base_cap_rate = 0.10
@gk_raiding_war_cap_rate_mult = 2
@gk_reaping_cap_rate = 100
@gk_max_pops_per_abduction = 100

####################
# Pop Cap Values   #
####################

# Combined effective taken (plundered + emancipated) for cap checks
# This = Country scope (the raidee)
gk_raidee_effective_taken = {
	base = 0
	modifier = {
		is_variable_set = gk_raidee_plundered
		add = gk_raidee_plundered
	}
	modifier = {
		is_variable_set = gk_raidee_emancipated
		add = gk_raidee_emancipated
	}
}

# Abduction cap: (total_pops + effective_taken) * rate - effective_taken
# This = raidee country, event_target:gk_raider = raider
# Parameter: RATE — a literal, @variable, or value:script_value
gk_abduction_cap = {
	base = 0
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		mode = add
	}
	add = value:gk_raidee_effective_taken
	mult = $RATE$
	subtract = value:gk_raidee_effective_taken
	ceiling = yes
	min = 0
}

# Surrender: flat base × capacity mult
# This = raidee, event_target:gk_raider = raider
gk_surrender_cap_rate = {
	base = @gk_surrender_base_cap_rate
	mult = event_target:gk_raider.value:gk_raiding_capacity_mult
}

# Invasion/bombardment: base × war-type mult × capacity mult
# This = raidee, event_target:gk_raider = raider
gk_invasion_cap_rate = {
	base = @gk_invasion_base_cap_rate
	modifier = {
		event_target:gk_raider = {
			any_war = {
				is_war_participant = { who = prevprev side = defenders }
				OR = {
					using_war_goal = { type = wg_plunder owner = prev }
					using_war_goal = { type = wg_gk_emancipation owner = prev }
					using_war_goal = { type = wg_gk_robo_liberation owner = prev }
				}
			}
		}
		mult = @gk_raiding_war_cap_rate_mult
	}
	modifier = {
		# Reaping war = effectively unlimited
		event_target:gk_raider = {
			any_war = {
				is_war_participant = { who = prevprev side = defenders }
				using_war_goal = { type = wg_gk_reaping owner = prev }
			}
		}
		mult = @gk_reaping_cap_rate  # 0.10 * 100 = 10 (>>1), takes all
	}
	mult = event_target:gk_raider.value:gk_raiding_capacity_mult
}

# Variable accessors for localisation
gk_forfeited_pops_value = {
	base = 0
	modifier = {
		is_variable_set = gk_forfeited_pops
		add = gk_forfeited_pops
	}
	floor = yes
}

gk_raidee_plundered_value = {
	base = 0
	modifier = {
		is_variable_set = gk_raidee_plundered
		add = gk_raidee_plundered
	}
	floor = yes
}

gk_total_pops_raided_value = {
	base = 0
	modifier = {
		is_variable_set = gk_total_pops_raided
		add = gk_total_pops_raided
	}
	floor = yes
}

####################
# Abduction Values #
####################

# How many pops to take in one abduction iteration
# This = pop_group scope
# event_target:gk_raidee = country with gk_remaining_cap variable
# Returns: min(pop_group_size, min(100, remaining_cap)) - always whole number
gk_pops_to_abduct = {
	base = 0
	add = trigger:pop_group_size
	# Clamp to remaining cap if it's less than pop_group size
	max = event_target:gk_raidee.gk_remaining_cap
	# No more than 100 at a time
	max = @gk_max_pops_per_abduction
}

# Raiding capacity multiplier (1 + modifier value)
# This = country scope (the raider)
# Used to multiply cap rates for pop raiding
gk_raiding_capacity_mult = {
	base = 1
	add = modifier:gk_raiding_capacity
}

####################
# Pillage Values   #
####################

# Pillage efficiency multiplier (1 + modifier value)
# This = country scope (the raider)
# Used to multiply resources gained from pillaging
gk_pillage_efficiency_mult = {
	base = 1
	add = modifier:gk_pillage_efficiency
}

####################
# Pillage Progress  #
####################

# Devastation curve for pillage progress chance modifier
# This = planet scope
# Returns: (devastation/100)^2 * 100
# Used to adjust pillage speed based on planet devastation
gk_devastation_curve = {
	base = 0
	add = trigger:planet_devastation
	divide = 100
	pow = 2
	mult = 100
}

####################
# Resource Calcs   #
####################

# Base resource production per resource type
# These get multiplied by tier_months and loot_multiplier in scripted_effects
# This = planet scope
# Parameters: RESOURCE - the resource type to check production for
# Usage: value:gk_base_loot_value|RESOURCE|energy|
gk_base_loot_value = {
	base = 0
	complex_trigger_modifier = {
		trigger = planet_resource_compare
		parameters = {
			resource = $RESOURCE$
			type = produces
		}
		mode = add
	}
	floor = yes
}

####################
# Tooltip Values   #
####################

# Estimated pops to be taken on surrender
# This = country (defender/raidee)
# Root = country (raider)
# Display value: rate × 100 = 10 (base percentage)
gk_estimated_surrender_pops = {
	base = 0
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		mode = add
	}
	add = value:gk_raidee_effective_taken
	mult = @gk_surrender_base_cap_rate
	mult = root.value:gk_raiding_capacity_mult
	subtract = value:gk_raidee_effective_taken
	ceiling = yes
	min = 0
}

# Bombardment/invasion cap percentage for display
# This = country (raider)
# Display value: base rate × 100 × raiding_war_cap_rate_mult = 20 (base percentage)
gk_bombardment_cap_percent = {
	base = 20
	mult = value:gk_raiding_capacity_mult
	floor = yes
}

# Surrender cap percentage for display
# This = country (raider)
# Display value: base rate × 100 = 10 (base percentage)
gk_surrender_cap_percent = {
	base = 10
	mult = value:gk_raiding_capacity_mult
	floor = yes
}
