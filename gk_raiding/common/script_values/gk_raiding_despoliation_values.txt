# floor(loser's ACQUIRED specimens / 9), min 1 — used by gk_steal_specimens on a surrender. Evaluated on
# the raider; reads the loser's count via trigger_scope. Uses acquired_specimen_count (count of ACQUIRED
# specimens — count_exhibit would undercount, since not all acquired specimens are exhibited).
# NOTE (verify in-game): trigger_scope = event_target and steal_specimens' count = value: ; if error.log
# objects, fall back to a one-shot exported variable + a `while` loop.
gk_specimen_steal_count = {
	base = 0
	complex_trigger_modifier = {
		trigger = acquired_specimen_count
		trigger_scope = event_target:gk_raidee
		mode = add
	}
	divide = 9
	floor = yes
	min = 1
}
