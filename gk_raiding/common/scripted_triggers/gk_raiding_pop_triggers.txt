# GK Raiding Pop Scripted Triggers

#############################
# Raiding Bombardment Check #
#############################
gk_can_use_raiding_bombardment = {
	owner = {
		OR = {
			#enabled resource raiding for privateers
			has_valid_civic = civic_crusader_spirit_corporate

			#new raiding civics
			has_valid_civic = civic_gk_emancipators
			has_valid_civic = civic_gk_emancipators_megacorp
			has_valid_civic = civic_gk_robo_liberators
			has_valid_civic = civic_gk_robo_liberators_megacorp
			has_valid_civic = civic_gk_robo_liberators_machine
			has_valid_civic = civic_gk_barbaric_despoilers_machine
			has_valid_civic = civic_gk_barbaric_despoilers_megacorp
		}
	}
}

# CB Access Checks #
####################
# This = country scope
# Can use standard raiding CB (from civics, ascension perks, etc.)
gk_can_use_raiding_cb = {
	NOT = { has_policy_flag = gk_focus_no_pop_raiding }
	OR = {
		has_valid_civic = civic_barbaric_despoilers
		has_ascension_perk = ap_nihilistic_acquisition
		has_valid_civic = civic_gk_emancipators
		has_valid_civic = civic_gk_emancipators_megacorp
		has_valid_civic = civic_gk_robo_liberators
		has_valid_civic = civic_gk_robo_liberators_megacorp
		has_valid_civic = civic_gk_robo_liberators_machine
		has_valid_civic = civic_gk_barbaric_despoilers_machine
		has_valid_civic = civic_gk_barbaric_despoilers_megacorp
	}
}

# This = country scope
# Personal pop-raiding capability.
gk_can_pop_raid_personally = {
	gk_can_use_raiding_cb = yes
}

# This = country scope
# Can use Emancipation war goal (requires emancipation policy)
gk_can_use_emancipation_cb = {
	has_policy_flag = gk_focus_emancipation
}

# This = country scope
# Can use Robo-Liberation war goal (requires robo-liberation policy)
gk_can_use_robo_liberation_cb = {
	has_policy_flag = gk_focus_robo_liberation
}

# This = country scope
# Can use Reaping war goal (requires despoiler civic)
gk_can_use_reaping_cb = {
	OR = {
		has_valid_civic = civic_barbaric_despoilers
		has_valid_civic = civic_gk_barbaric_despoilers_machine
		has_valid_civic = civic_gk_barbaric_despoilers_megacorp
	}
}

# War Goal Checks  #
####################
# This = country scope
gk_has_robo_liberation_war = {
	any_war = {
		OR = {
			AND = {
				is_war_participant = { who = prev side = attackers }
				using_war_goal = {
					type = wg_gk_robo_liberation
					owner = attacker
				}
			}
			AND = {
				is_war_participant = { who = prev side = defenders }
				using_war_goal = {
					type = wg_gk_robo_liberation
					owner = defender
				}
			}
		}
	}
}

# War-type specific "war with" triggers
# This = country scope (defender/raidee)
# Root = country scope (attacker/raider)
gk_has_standard_raiding_war_with = {
	root = {
		any_war = {
			OR = {
				AND = {
					is_war_participant = { who = prev side = attackers }
					is_war_participant = { who = prevprev side = defenders }
					using_war_goal = { type = wg_plunder owner = attacker }
				}
				AND = {
					is_war_participant = { who = prev side = defenders }
					is_war_participant = { who = prevprev side = attackers }
					using_war_goal = { type = wg_plunder owner = defender }
				}
			}
		}
	}
}

# Pop Group Checks #
####################
# Root = hostile country (attacker/raider)
# This = pop_group
gk_can_abduct_pop_group_war_goal_check = {
	# Only allow pop abduction in explicit raiding contexts.
	# Non-raiding wars (conquest, subjugation, etc.) do NOT grant pop capture.
	OR = {
		# No formal war with owner (primitives, first contact, etc.) - allow
		planet = {
			owner = {
				is_primitive = yes
			}
		}
		# Total war - always allow
		planet = {
			owner = {
				root = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = {
									who = prev
									side = attackers
								}
								is_war_participant = {
									who = prevprev
									side = defenders
								}
							}
							AND = {
								is_war_participant = {
									who = prev
									side = defenders
								}
								is_war_participant = {
									who = prevprev
									side = attackers
								}
							}
						}
						is_total_war = yes
					}
				}
			}
		}
		# Standard raiding or reaping war with this defender - any pop
		planet = {
			owner = {
				OR = {
					gk_has_standard_raiding_war_with = yes
					gk_has_reaping_war_with = yes
				}
			}
		}
		# Emancipation war with this defender - enslaved only
		AND = {
			planet = { owner = { gk_has_emancipation_war_with = yes } }
			is_enslaved = yes
		}
		# Robo-liberation war with this defender - robots only
		AND = {
			planet = { owner = { gk_has_robo_liberation_war_with = yes } }
			is_robotic_species = yes
		}
	}
}

# Root = hostile country (attacker/raider)
# This = pop_group
# Flag-based cap check - all rate logic is in gk_update_raid_cap_flag
gk_can_abduct_pop_group_cap_check = {
	OR = {
		# Total war - no cap
		planet = {
			owner = {
				root = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = {
									who = prev
									side = attackers
								}
								is_war_participant = {
									who = prevprev
									side = defenders
								}
							}
							AND = {
								is_war_participant = {
									who = prev
									side = defenders
								}
								is_war_participant = {
									who = prevprev
									side = attackers
								}
							}
						}
						is_total_war = yes
					}
				}
			}
		}
		# Primitives - no cap
		planet = {
			owner = {
				is_primitive = yes
			}
		}
		# Reaping war - no cap
		planet = {
			owner = {
				gk_has_reaping_war_with = yes
			}
		}
		# Cap not reached (flag-based check)
		planet = {
			owner = {
				root = {
					any_war = {
						OR = {
							AND = {
								is_war_participant = { who = prev side = attackers }
								is_war_participant = { who = prevprev side = defenders }
								OR = {
									using_war_goal = { type = wg_plunder owner = attacker }
									using_war_goal = { type = wg_gk_emancipation owner = attacker }
									using_war_goal = { type = wg_gk_robo_liberation owner = attacker }
								}
								prevprev = {
									NOT = { has_country_flag = gk_raid_cap_reached_by_@attacker }
								}
							}
							AND = {
								is_war_participant = { who = prev side = defenders }
								is_war_participant = { who = prevprev side = attackers }
								OR = {
									using_war_goal = { type = wg_plunder owner = defender }
									using_war_goal = { type = wg_gk_emancipation owner = defender }
									using_war_goal = { type = wg_gk_robo_liberation owner = defender }
								}
								prevprev = {
									NOT = { has_country_flag = gk_raid_cap_reached_by_@defender }
								}
							}
						}
					}
				}
			}
		}
	}
}

# Root = hostile country (attacker/raider)
# This = pop_group
gk_can_abduct_pop_group_robot_check = {
	# Check: Robot raiding policy and robo-liberation war restrictions
	OR = {
		# Robo-liberators civic OR in robo-liberation war - ONLY robots
		AND = {
			OR = {
				root = {
					has_valid_civic = civic_gk_robo_liberators
				}
				root = {
					has_valid_civic = civic_gk_robo_liberators_megacorp
				}
				root = {
					has_valid_civic = civic_gk_robo_liberators_machine
				}
				root = {
					gk_has_robo_liberation_war = yes
				}
			}
			is_robotic_species = yes
		}
		# Robots only policy - ONLY robots
		AND = {
			root = {
				has_policy_flag = gk_robots_only_raiding
			}
			is_robotic_species = yes
		}
		# Forbid robot raiding - ONLY organics
		AND = {
			root = {
				has_policy_flag = gk_forbid_robot_raiding
			}
			is_robotic_species = no
		}
		# Allow robot raiding (and not robo-liberator/robo-lib war) - both allowed
		AND = {
			root = {
				has_policy_flag = gk_allow_robot_raiding
				NOR = {
					has_valid_civic = civic_gk_robo_liberators
					has_valid_civic = civic_gk_robo_liberators_megacorp
					has_valid_civic = civic_gk_robo_liberators_machine
				}
				NOT = {
					gk_has_robo_liberation_war = yes
				}
			}
		}
	}
}

# Root = hostile country (attacker/raider)
# This = pop_group
gk_can_abduct_pop_group = {
	# Vanilla checks the pop is eligible to be raided in the first place.
	# This filters based on the raiding quotas and war goal restrictions.
	# Check 0: No Pop-Raiding policy blocks all pop raiding
	root = {
		gk_can_pop_raid_personally = yes
	}
	# Check 1: War Goal Restrictions
	gk_can_abduct_pop_group_war_goal_check = yes
	# Check 2: Per-Raidee Limit
	gk_can_abduct_pop_group_cap_check = yes
	# Check 3: Robot Raiding Policy
	gk_can_abduct_pop_group_robot_check = yes
}

# This = planet scope
# Check if planet is good destination for raided pops
gk_is_valid_pop_destination = {
	is_colony = yes
	exists = owner
	free_housing >= 1
	free_amenities >= 1
}
