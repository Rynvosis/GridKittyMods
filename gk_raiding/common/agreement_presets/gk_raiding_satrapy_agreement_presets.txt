# Universal warrior-pact preset for the satrapy specialist subject type.
# Available to any overlord (mirrors preset_bulwark gating). Vanilla
# preset_satrapy / preset_dominion are untouched — this is additive.

# Re-declare the constants we reference. Vanilla scripted_variables don't
# cross file boundaries — without these we'd get "Variable not set" errors.
@protectorate_tech_threshold = 0.4
@specialist_base_acceptance_value = 50
@specialist_aligned_ethics_value = 50
@specialist_unaligned_ethics_value = -50

preset_satrapy_warrior = {
	icon = "GFX_diplomacy_status_is_satrapy"
	specialist_type = specialist_satrapy
	term_data = {
		resource_terms = {
			# Overlord subsidizes subject 15% basics — partial offset for the
			# tier penalties (which hit consumer goods + unity + naval cap).
			{
				key = resource_subsidies_basic
				value = -0.15
			}
		}
		discrete_terms = {
			{
				key = specialist_type
				value = specialist_satrapy
			}
			{
				key = subject_integration
				value = subject_can_not_be_integrated
			}
			{
				key = subject_diplomacy
				value = subject_can_do_diplomacy
			}
			{
				key = subject_expand
				value = subject_can_expand
			}
			{
				key = joins_overlord_wars
				value = joins_overlord_wars_all
			}
			{
				key = joins_subject_wars
				value = joins_subject_wars_all
			}
			{
				key = subject_holdings_limit
				value = subject_holdings_limit_1
			}
			{
				key = subject_loyalty
				value = subject_loyalty_effects
			}
			{
				key = protectorate
				value = subject_is_protectorate
			}
		}
	}

	overlord_weight = {
		base = @specialist_base_acceptance_value
		modifier = {
			add = @specialist_aligned_ethics_value
			is_militarist = yes
		}
		modifier = {
			factor = 0.5
			has_specialist_subject_type = { TYPE = satrapy }
			NOT = { FROM = { is_specialist_subject_type = { TYPE = satrapy } } }
		}
		modifier = {
			factor = 1.5
			FROM = { is_specialist_subject_type = { TYPE = satrapy } }
		}
		modifier = {
			factor = 1.2
			num_rivals > 0
		}
	}

	subject_weight = {
		base = 25
		modifier = {
			add = @specialist_aligned_ethics_value
			is_militarist = yes
		}
		modifier = {
			add = @specialist_unaligned_ethics_value
			is_pacifist = yes
		}
	}

	potential = {
		has_overlord_dlc = yes
		NOR = {
			has_civic = civic_inwards_perfection
			FROM = { has_civic = civic_inwards_perfection }
		}
		OR = {
			is_specialist_subject_type = { TYPE = satrapy }
			has_country_flag = specialistion_subject_conversion_flag
			tech_unlocked_ratio = {
				who = from
				ratio >= @protectorate_tech_threshold
			}
			# Khan-style overlords can always convert their satrapies regardless
			# of subject tech parity — the marauder horde doesn't care.
			FROM = {
				OR = {
					is_country_type = awakened_marauders
					has_country_flag = gk_utils_spawned_marauder
				}
			}
		}
	}
}
