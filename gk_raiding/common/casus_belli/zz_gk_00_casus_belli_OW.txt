# Overwrite: cb_despoliation (plunder + reaping war goals)
# Overwrite: cb_pirate_raid (resource plundering)

# Pop raiding CB — plunder and reaping wars
# Access gated by raiding focus policy
cb_despoliation = {
	show_in_diplomacy = no
	show_notification = no

	potential = {
		is_country_type = default
		OR = {
			has_policy_flag = gk_focus_raiding
			has_policy_flag = gk_focus_reaping
		}
	}

	is_valid = {
		has_total_war_cb = no
		NOT = { is_overlord_to = FROM }
		OR = {
			is_subject = no
			overlord = { NOT = { is_same_value = from } }
		}
	}

	proxy_war_resources = {
		category = proxy_war
		cost = {
			influence = 100
		}
	}
	on_proxy_war_start = {
		pay_proxy_war_additional_resources = { VALUE = -25 }
	}
}

# Emancipation CB — targets enslaved organic pops
# Used by wg_gk_emancipation
cb_gk_emancipation = {
	show_in_diplomacy = no
	show_notification = no

	potential = {
		is_country_type = default
		has_policy_flag = gk_focus_emancipation
	}

	is_valid = {
		has_total_war_cb = no
		NOT = { is_overlord_to = FROM }
		OR = {
			is_subject = no
			overlord = { NOT = { is_same_value = from } }
		}
		# Target must have enslaved pops
		FROM = {
			any_owned_pop_group = {
				is_enslaved = yes
			}
		}
	}

	proxy_war_resources = {
		category = proxy_war
		cost = {
			influence = 100
		}
	}
	on_proxy_war_start = {
		pay_proxy_war_additional_resources = { VALUE = -25 }
	}
}

		}
	}
