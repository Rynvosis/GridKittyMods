# EMANCIPATORS     #
####################
# Egalitarian civic focused on "liberating" enslaved pops
# Grants access to Emancipation policy/war goal and bonus efficiency
civic_gk_emancipators = {
	playable = { always = yes }
	ai_playable = { always = yes }

	description = civic_gk_emancipators_effects

	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } }
		authority = { NOT = { value = auth_corporate } }
	}

	possible = {
		ethics = {
			OR = {
				text = civic_tooltip_egalitarian
				value = ethic_egalitarian
				value = ethic_fanatic_egalitarian
			}
			OR = {
				text = civic_tooltip_militarist
				value = ethic_militarist
				value = ethic_fanatic_militarist
			}
		}
	}

	random_weight = {
		base = 5
		modifier = {
			factor = 2
			has_ethic = ethic_fanatic_egalitarian
		}
	}

	modifier = {
		army_morale = 0.50
	}

	alternate_civic_version = civic_gk_emancipators_megacorp

	modification = yes

	# Grants access to Emancipation policy option via potential check
	# The policy itself grants the base modifiers
}

# REAVERS (Standard)         #
##############################
# Normal-empire resource-raider. Native plundering + Plunderer trait + (Overlord) +1 enclave capacity.
# Mirrors vanilla Letters of Marque (civic_crusader_spirit_corporate) for the megacorp side.
civic_gk_reavers = {
	playable = { always = yes }
	ai_playable = { always = yes }

	description = civic_gk_reavers_effects

	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } }
		authority = { NOT = { value = auth_corporate } }
	}

	possible = {
		is_nomadic = no # Nomad empires use Void Reavers instead.
		ethics = {
			OR = {
				text = civic_tooltip_authoritarian_or_militarist
				value = ethic_militarist
				value = ethic_fanatic_militarist
				value = ethic_authoritarian
				value = ethic_fanatic_authoritarian
			}
			NOR = {
				text = civic_tooltip_not_pacifist
				value = ethic_pacifist
				value = ethic_fanatic_pacifist
			}
		}
	}

	random_weight = {
		base = 5
		modifier = {
			factor = 2
			has_ethic = ethic_fanatic_militarist
		}
	}

	modifier = {
		ships_upkeep_mult = -0.05
	}

	# With Overlord, the civic ALSO grants +1 enclave capacity. swap_type modifiers stack on top of the
	# base modifier (they don't replace it), so the base -5% ship upkeep persists — mirrors vanilla
	# Letters of Marque, whose Overlord swap_type likewise lists only the enclave-capacity delta.
	swap_type = {
		trigger = {
			host_has_dlc = "Overlord"
		}
		description = civic_gk_reavers_effects
		modifier = {
			country_enclave_capacity_add = 1
		}
	}

	modification = yes
}
