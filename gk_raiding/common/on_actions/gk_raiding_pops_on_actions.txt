# GK Raiding Pops On Actions

# A war is beginning, executed for every country in the war
# Root = Country
# From = War
on_war_beginning = {
	events = {
		gk_raiding_pops.400  # Reaping war declaration opinion
	}
}

# Pop Group is abducted by raiding stance
# This = Pop Group scope
# From = planet abducted from
on_pop_group_abducted = {
	events = {
		gk_raiding_pops.100  # Track raided pops
	}
}

# Planet has taken damage from orbital bombardment - Damage is applied daily
# This = Planet
# From = Bombarder (country)
on_planet_bombarded = {
	events = {
		gk_raiding_pops.200  # Pop-raiding forced surrender resolution
	}
}

# Attackers win ground combat (before controller switches)
# This = Attacking Country
# From = Defending Country
# FromFrom = Planet
on_planet_attackers_win = {
	events = {
		gk_raiding_pops.210  # Pop landing - instant pop raid
	}
}

# Planet changes hands through aggressive conquest (after ownership switches)
# This = Planet
# From = New Owner
# FromFrom = Former Owner
on_planet_conquer = {
	events = {
		gk_raiding_pops.220  # Pop forced-surrender cleanup
	}
}

# Yearly pulse for all countries
# This = Country
on_yearly_pulse_country = {
	events = {
		gk_raiding_pops.110  # Decay raidee raided pops
		gk_raiding_pops.120  # Cleanup raid trackers and cap flags
	}
}
