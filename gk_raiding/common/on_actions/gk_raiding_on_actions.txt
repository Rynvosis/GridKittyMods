# GK Raiding On Actions

# Setup machine despoilers secondary species
on_game_start_country = {
	events = {
		gk_raiding.2
	}
}

# A war is beginning, executed for every country in the war
# Root = Country
# From = War
on_war_beginning = {
	events = {
		gk_raiding.800  # Reaping war declaration opinion
	}
}

# A war has ended
# Root = Loser
# From = Main Winner
on_war_ended = {
	events = {
		gk_raiding.53  # Cleanup invasion pillage cashout flags
	}
}

# Pop Group is abducted by raiding stance
# This = Pop Group scope
# From = planet abducted from
on_pop_group_abducted = {
	events = {
		gk_raiding.101  # Track raided pops
	}
}

# Planet has taken damage from orbital bombardment - Damage is applied daily
# This = Planet
# From = Bombarder (country)
on_planet_bombarded = {
	events = {
		gk_raiding.100  # Resource pillage from devastation + forced surrender resolution
	}
}

# Planet has taken damage from ground combat - Damage is applied daily
# This = Planet
# From = Damaging country
on_ground_combat_devastation = {
	events = {
		gk_raiding.100  # Resource pillage from devastation
	}
}

# Attackers win ground combat (before controller switches)
# This = Attacking Country
# From = Defending Country
# FromFrom = Planet
on_planet_attackers_win = {
	events = {
		gk_raiding.200  # Army landing - instant pillage + pop raid
	}
}

# Planet changes hands through aggressive conquest (after ownership switches)
# This = Planet
# From = New Owner
# FromFrom = Former Owner
on_planet_conquer = {
	events = {
		gk_raiding.201  # Orbital surrender - trigger the same invasion raid flow
	}
}

# Yearly pulse for all countries
# This = Country
on_yearly_pulse_country = {
	events = {
		gk_raiding.50  # Decay raidee raided pops
		gk_raiding.52  # Cleanup raid trackers and cap flags
	}
}
