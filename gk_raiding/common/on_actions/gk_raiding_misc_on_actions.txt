# GK Raiding Misc On Actions

# Setup machine despoilers secondary species
on_game_start_country = {
	events = {
		gk_raiding_misc.1
	}
}

on_added_to_leader_pool = {
	events = {
		gk_raiding_misc.110  # Acquisitive Officers tradition → Plunderer + Corsair/Shipbreaker
		gk_raiding_misc.114  # Resource-raid civics → Plunderer only
	}
}

on_ship_destroyed_perp = {
	events = {
		gk_raiding_misc.111   # Acquisitive Officers ship-kill trait top-up
		# Privateer + Corsair ship STEALING now uses vanilla's boarding-cables modifier
		# (ship_steal_defeated_ship_chance on the raider / civic) instead of custom on-kill events.
	}
}

on_leader_hired = {
	events = {
		gk_raiding_misc.112  # Acquisitive Officers tradition → Plunderer + Corsair/Shipbreaker
		gk_raiding_misc.116  # Resource-raid civics → Plunderer only
	}
}
