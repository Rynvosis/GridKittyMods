# Destructive Engineering — destruction as an applied discipline
# Explosive ordnance, bombardment, and collateral wrecking
# Adds tech_missiles_2 as a research option at game start (see gk_ec_civic.500)

civic_gk_ec_destructive_engineering = {
	playable = { always = yes }
	ai_playable = { always = yes }

	potential = {
		ethics = { NOT = { value = ethic_gestalt_consciousness } }
		authority = { NOT = { value = auth_corporate } }
	}

	possible = {
		ethics = {
			OR = {
				text = civic_tooltip_militarist
				value = ethic_militarist
				value = ethic_fanatic_militarist
			}
		}
	}

	random_weight = { base = @civic_default_random_weight }
	ai_weight = {
		base = @ai_civic_default_base_weight
		modifier = {
			factor = @ai_civic_personality_match_factor
			OR = {
				has_ai_personality = hegemonic_imperialists
				has_ai_personality = honorbound_warriors
			}
		}
	}

	description = civic_gk_ec_destructive_engineering_effects

	modifier = {
		weapon_type_explosive_weapon_damage_mult = 0.10
		ship_orbital_bombardment_mult = 0.33
		army_collateral_damage_mult = 0.25
	}

	alternate_civic_version = civic_gk_ec_destructive_engineering_megacorp

	modification = yes
}
