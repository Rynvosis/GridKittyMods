# this is brute force AS FUCK
#  basically given the target system it will look for systems within a 200 radius
#  then check every hyperlane in that same radius against the intersect hyperlane trigger.
rec_add_every_non_intersecting_hyperlane = {
	save_event_target_as = rec_system_a
	every_neighbor_system_euclidean = {
		limit = {
			NOT = { has_hyperlane_to = event_target:rec_system_a } 
		}

		save_event_target_as = rec_system_b
		create_variable_if_not_exists = {
			VARIABLE = rec_dAB
			VALUE_IF_CREATED = 0
		}
		set_variable = {
			which = rec_dAB
			value = value:rec_distance_between|FROM|rec_system_a|TO|rec_system_b|
		}

		#now we need to check system C
		every_neighbor_system_euclidean = {
			limit = {
				NOR = {
					is_same_value = event_target:rec_system_a
					is_same_value = event_target:rec_system_b
				}
			}

			save_event_target_as = rec_system_c
			create_variable_if_not_exists = {
				VARIABLE = rec_dAC
				VALUE_IF_CREATED = 0
			}
			export_trigger_value_to_variable = {
				trigger = distance
				variable = rec_dAC
				parameters = { source = event_target:rec_system_a type = euclidean }
			}
			create_variable_if_not_exists = {
				VARIABLE = rec_dBC
				VALUE_IF_CREATED = 0
			}
			export_trigger_value_to_variable = {
				trigger = distance
				variable = rec_dBC
				parameters = { source = event_target:rec_system_b type = euclidean }
			}

			#check every hyperlane from C to get D
			every_neighbor_system = {
				limit = {
					NOR = { 
						# has_star_flag = rec_hyperlane_check_@event_target:rec_system_a_@event_target:rec_system_b
						has_hyperlane_to = event_target:rec_system_a
						is_same_value = event_target:rec_system_a
						has_hyperlane_to = event_target:rec_system_b
						is_same_value = event_target:rec_system_b
					}
				}
				
				save_event_target_as = rec_system_d

				create_variable_if_not_exists = {
					VARIABLE = rec_dAD
					VALUE_IF_CREATED = 0
				}
				export_trigger_value_to_variable = {
					trigger = distance
					variable = rec_dAD
					parameters = { source = event_target:rec_system_a type = euclidean }
				}
				create_variable_if_not_exists = {
					VARIABLE = rec_dBD
					VALUE_IF_CREATED = 0
				}
				export_trigger_value_to_variable = {
					trigger = distance
					variable = rec_dBD
					parameters = { source = event_target:rec_system_b type = euclidean }
				}
				create_variable_if_not_exists = {
					VARIABLE = rec_dCD
					VALUE_IF_CREATED = 0
				}
				export_trigger_value_to_variable = {
					trigger = distance
					variable = rec_dCD
					parameters = { source = event_target:rec_system_c type = euclidean }
				}

				#copy variables over from previous scopes:
				# set_variable = {
				# 	which = rec_dAB
				# 	value = value:rec_distance_between|FROM|rec_system_a|TO|rec_system_b|
				# }
				# set_variable = {
				# 	which = rec_dAC
				# 	value = value:rec_distance_between|FROM|rec_system_a|TO|rec_system_c|
				# }
				# set_variable = {
				# 	which = rec_dBC
				# 	value = value:rec_distance_between|FROM|rec_system_b|TO|rec_system_c|
				# }
			}
		}
	}
}

rec_calculate_PQ_relative_abs_RS_distance = {
	# do systems R and S fall on opposite sides of P-Q?
	set_variable = {
		which = rec_xR
		value = value:rec_x_from_lengths|AB|$dPQ$|AC|$dPR$|BC|$dQR$|
	}
	set_variable = {
		which = rec_yR
		value = value:rec_abs_y_from_x|HYPOTENUSE|$dPR$|X|rec_xQ|
	}

	set_variable = {
		which = rec_xS
		value = value:rec_x_from_lengths|AB|$dPQ$|AC|$dPS$|BC|$dQS$|
	}
	set_variable = {
		which = rec_yS
		value = value:rec_abs_y_from_x|HYPOTENUSE|$dPS$|X|rec_xS|
	}

	#if distance is close enough when both are positive, its likely they both lie on same side
	set_variable = {
		which = $distance$
		value = value:rec_distance|X1|rec_xR|Y1|rec_yR|X2|rec_xS|Y2|rec_yS|
	}

	clear_variable = rec_xR
	clear_variable = rec_yR
	clear_variable = rec_xS
	clear_variable = rec_yS
}
