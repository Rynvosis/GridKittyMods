# Stratified Faction
##############################
rec_stratified = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_stratified
	unique = yes
	support_multiplier = 1
	visual_ethic = "rec_stratified_visual_ethic"

	resources = {
		category = pop_factions
		produces = {
			trigger = {
				owner = {
					OR = {
						years_passed > 10
						AND = {
							days_passed > 30
							has_valid_civic = civic_parliamentary_system
						}
						AND = {
							days_passed > 350
							has_origin = origin_broken_shackles
						}
					}
				}
			}
			unity = @faction_base_output
		}
	}

	extortion_resource = {
		category = pop_factions
		produces = {
			energy = @faction_base_output
			mult = @faction_extortion_energy_ratio
		}
	}

	is_potential = {
		is_faction_allowed = yes
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		can_think = yes
	}

	attraction = {
		base = 150
		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { is_ruler = yes }
			}
		}
		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
		modifier = {
			factor = 0.25
			NOT = { is_same_species = owner_species }
		}
		modifier = {
			factor = 0
			planet = { has_modifier = planet_culture_shock }
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			is_event_leader = yes
		}
	}

	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"
		fulfilled_effect = -20
		unfulfilled_effect = -0.001
		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_rec_stratified_outraged_gc
			}
		}
		trigger = {
			owner = {
				has_country_flag = faction_rec_stratified_outraged_gc
			}
		}
	}

	#TODO: faction-specific demands

	# Faction Demands for Ruler and Council positions.
	inline_script = {
		script = "paragon/global_faction_demands"
		ETHIC = ethic_stratified
		ETHIC_FANATIC = ethic_fanatic_stratified
		ETHIC_OPPOSED = ethic_pluralist
	}

	on_create = {
		pop_faction_event = { id = factions.0 }
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_stratified
			}
			remove_modifier = promoted_stratified
		}
		if = {
			limit = {
				has_modifier = suppressed_stratified
			}
			remove_modifier = suppressed_stratified
		}
	}

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"
			cost = {
				unity = 5000
			}
			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { any_agreement = { agreement_preset = preset_dominion } }
					}
				}
			}
			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = {
							has_modifier = embraced_faction_timer
						}
					}
				}
				support > 0.2
				parameter:empire = {
					NOT = {
						has_ethic = "ethic_fanatic_stratified"
					}
				}
			}
			effect = {
				add_modifier = {
					modifier = embraced_faction
					days = 3600
				}
				parameter:empire = {
					shift_ethic = ethic_stratified
					hidden_effect = {
						add_modifier = {
							modifier = embraced_faction_timer
							days = 3600
						}
						every_pop_faction = {
							limit = {
								NOT = {
									is_same_value = root
								}
							}
							add_modifier = {
								modifier = embraced_another_faction
								days = 3600
							}
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default
								has_communications = prev
								has_intel_level = {
									who = prev
									category = government
									level >= 1
								}
							}
							country_event = {
								id = factions.2000
							}
						}
					}
				}
			}
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.5
					owner = {
						has_ethic = ethic_stratified
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_faction = {
							count < 4
						}
					}
				}
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_stratified
						has_modifier = promoted_stratified
					}
				}
			}
			effect = {
				parameter:empire = {
					add_modifier = {
						modifier = promoted_stratified
						days = -1
					}
				}
			}
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_stratified
				}
			}
			effect = {
				parameter:empire = {
					remove_modifier = promoted_stratified
				}
			}
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_stratified
						has_modifier = promoted_stratified
					}
				}
			}
			effect = {
				add_modifier = {
					modifier = suppressed_faction
					days = -1
				}
				parameter:empire = {
					add_modifier = {
						modifier = suppressed_stratified
						days = -1
					}
				}
			}
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_stratified
				}
			}
			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_stratified
				}
			}
			ai_weight = {
				base = 0
			}
		}
		# Faction Extortion Buttons
		extort_faction = {
			title = "EXTORT_FACTION"
			description = "EXTORT_FACTION_DESC"
			potential = {
				exists = owner
				owner = {
					is_oligarchic_authority = yes
					has_country_flag = cyber_collectivist
				}
				is_faction_extorted = no
			}
			effect = {
				set_faction_extorted = yes
				custom_tooltip = energy_extortion_tt
				custom_tooltip = ONLY_ONE_EXTORTED_FACTION_TT
			}
			ai_weight = {
				base = 0
			}
		}
		cancel_extort_faction = {
			title = "CANCEL_EXTORT_FACTION"
			description = "CANCEL_EXTORT_FACTION_DESC"
			potential = {
				exists = owner
				owner = {
					is_oligarchic_authority = yes
					has_country_flag = cyber_collectivist
				}
				is_faction_extorted = yes
			}
			effect = {
				set_faction_extorted = no
			}
			ai_weight = {
				base = 0
			}
		}
	}
}
