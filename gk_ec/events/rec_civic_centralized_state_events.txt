namespace = rec_civic_centralized_state

# gamestart, add heart_of_the_empire modifier to capital
country_event = {
	id = rec_civic_centralized_state.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			has_valid_civic = civic_rec_centralized_state
			has_valid_civic = civic_rec_centralized_state_hive
			has_valid_civic = civic_rec_centralized_state_machine
			has_valid_civic = civic_rec_centralized_state_corporate
		}
	}

	immediate = { 
		if = {
			limit = {
				OR = {
					has_valid_civic = civic_rec_centralized_state
					has_valid_civic = civic_rec_centralized_state_corporate
				}
			}
			capital_scope = { add_building = building_rec_centralized_state_capital }
		}
		else_if = {
			limit = {
				OR = {
					has_valid_civic = civic_rec_centralized_state_hive
					has_valid_civic = civic_rec_centralized_state_machine
				}
			}
			capital_scope = { add_building = building_rec_centralized_state_capital_gestalt }
		}
	}
}

# Resettle colonists to capital:
planet_event = {
	id = rec_civic_centralized_state.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_building = building_rec_centralized_state_colony

		exists = owner
		owner = { 
			exists = capital_scope
			capital_scope = { OR = {
				has_building = building_rec_centralized_state_capital
				has_building = building_rec_centralized_state_capital_gestalt	
			} }
		}
	}

	immediate = {
		#resettle 10% of civilians to capital
		every_owned_pop_group = {
			limit = {
				OR = {
					is_pop_category = civilian
					is_unemployed = yes
				}
				has_migration_control = { type = no }
			}
			resettle_pop_group = {
				POP_GROUP = this
				PLANET = owner.capital_scope
				PERCENTAGE = 0.1
			}
		}
	}
}
